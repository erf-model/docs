140 void z0_est(Real z, Real bflx, Real wnd, Real ustar, Real &z0) {
141  Real vonk = 0.4;
142  Real eps = 1.0e-10;
143  Real am = 4.8;
144  Real bm = 19.3;
145  Real c1 = 3.14159/2.0 - 3.0*log(2.0);
146  Real rlmo = -bflx*vonk/(ustar*ustar*ustar+eps);
147  Real zeta = min(1.0,z*rlmo);
148  Real x;
149  Real psi1;
150  if(zeta >= 0.0) {
151  psi1 = -am*zeta;
152  }
153  else {
154  x = sqrt(sqrt(1.0-bm*zeta));
155  psi1 = 2.0*log(1.0+x) + log(1.0+x*x) -2.0*atan(x) + c1;
156  }
157  Real lnz = max(0.0, vonk*wnd/(ustar+eps) +psi1);
158  z0 = z*exp(-lnz);
159 }
