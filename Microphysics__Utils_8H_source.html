138 void z0_est(amrex::Real z, amrex::Real bflx, amrex::Real wnd, amrex::Real ustar, amrex::Real &z0) {
139  amrex::Real vonk = 0.4;
140  amrex::Real eps = 1.0e-10;
141  amrex::Real am = 4.8;
142  amrex::Real bm = 19.3;
143  amrex::Real c1 = 3.14159/2.0 - 3.0*log(2.0);
144  amrex::Real rlmo = -bflx*vonk/(ustar*ustar*ustar+eps);
145  amrex::Real zeta = std::min(1.0,z*rlmo);
146  amrex::Real x;
147  amrex::Real psi1;
148  if(zeta >= 0.0) {
149  psi1 = -am*zeta;
150  }
151  else {
152  x = std::sqrt(sqrt(1.0-bm*zeta));
153  psi1 = 2.0*std::log(1.0+x) + std::log(1.0+x*x) -2.0*std::atan(x) + c1;
154  }
155  amrex::Real lnz = std::max(0.0, vonk*wnd/(ustar+eps) +psi1);
156  z0 = z*std::exp(-lnz);
157 }

171  amrex::Real qss = qploc-qgg;
172  term_vel = (omg*vgrau*std::pow(rho*qgg,cgrau) + (1.0-omg)*vsnow*std::pow(rho*qss,csnow));
173  }
174  else {

176  amrex::Real qrr = omp*qploc;
177  amrex::Real qss = qploc-qrr;
178  amrex::Real qgg = omg*qss;

181  }
