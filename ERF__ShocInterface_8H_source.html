118  // Most individual processes have a pre-processing step that constructs needed variables from
119  // the set of fields stored in the field manager. A structure like this defines those operations,
120  // which can then be called during run_impl in the main .cpp code.
121  // Structure to handle the local generation of data needed by shoc_main in run_impl
122  struct SHOCPreprocess {
123  SHOCPreprocess () = default;

125  KOKKOS_INLINE_FUNCTION
126  void operator()(const Kokkos::TeamPolicy<KT::ExeSpace>::member_type& team) const
127  {
128  const int i = team.league_rank();
129 
130  const Real zvir = C::ZVIR;
131  const Real cpair = C::Cpair;
132  const Real ggr = C::gravit;
133  const Real inv_ggr = 1/ggr;
134  const Real mintke = SC::mintke;
135 
136  const int nlev_packs = ekat::npack<Spack>(nlev);
137 
138  Kokkos::parallel_for(Kokkos::TeamVectorRange(team, nlev_packs), [&] (const Int& k)
139  {
140  cldfrac_liq_prev(i,k)=cldfrac_liq(i,k);
141 
142  // Inverse of Exner. In non-rel builds, assert that exner != 0 when in range before computing.
143  const Spack exner = PF::exner_function(p_mid(i,k));
144  const Smask nonzero = (exner != 0);
145  EKAT_KERNEL_ASSERT((nonzero || !(ekat::range<IntSmallPack>(k*Spack::n) < nlev)).all());
146  inv_exner(i,k).set(nonzero, 1/exner);
147 
148  tke(i,k) = ekat::max(mintke, tke(i,k));
149 
150  // Tracers are updated as a group. The tracers tke and qc act as separate inputs to shoc_main()
151  // and are therefore updated differently to the tracers group's monolithic field. Here, we make
152  // a copy if each of these tracers and pass to shoc_main() so that changes to the tracer group
153  // does not alter tke or qc values. Then during post processing, we copy back correct values of
154  // tke and qc to tracer group in postprocessing.
155  // TODO: remove *_copy views once SHOC can request a subset of tracers.
156  tke_copy(i,k) = tke(i,k);
157  qc_copy(i,k) = qc(i,k);
158 
159  qw(i,k) = qv(i,k) + qc(i,k);
160 
161  // Temperature
162  // NOTE: theta_v (thv) is intentionally different from one in HOMME
163  const auto theta_zt = PF::calculate_theta_from_T(T_mid(i,k),p_mid(i,k));
164  thlm(i,k) = PF::calculate_thetal_from_theta(theta_zt,T_mid(i,k),qc(i,k));
165  thv(i,k) = theta_zt*(1 + zvir*qv(i,k) - qc(i,k));
166 
167  // Vertical layer thickness
168  dz(i,k) = PF::calculate_dz(pseudo_density(i,k), p_mid(i,k), T_mid(i,k), qv(i,k));
169 
170  rrho(i,k) = inv_ggr*(pseudo_density(i,k)/dz(i,k));
171  wm_zt(i,k) = -1*omega(i,k)/(rrho(i,k)*ggr);
172  });
173  team.team_barrier();
174 
175  // Compute vertical layer heights
176  const auto dz_s = ekat::subview(dz, i);
177  const auto z_int_s = ekat::subview(z_int, i);
178  const auto z_mid_s = ekat::subview(z_mid, i);
179  PF::calculate_z_int(team,nlev,dz_s,z_surf,z_int_s);
180  team.team_barrier();
181  PF::calculate_z_mid(team,nlev,z_int_s,z_mid_s);
182  team.team_barrier();
183 
184  const int nlevi_v = nlev/Spack::n;
185  const int nlevi_p = nlev%Spack::n;
186  Kokkos::parallel_for(Kokkos::TeamVectorRange(team, nlev_packs), [&] (const Int& k)
187  {
188  zt_grid(i,k) = z_mid(i,k) - z_int(i, nlevi_v)[nlevi_p];
189  zi_grid(i,k) = z_int(i,k) - z_int(i, nlevi_v)[nlevi_p];
190 
191  // Dry static energy
192  shoc_s(i,k) = PF::calculate_dse(T_mid(i,k),z_mid(i,k),phis(i));
193 
194  if (k+1 == nlev_packs) zi_grid(i,nlevi_v)[nlevi_p] = 0;
195  });
196  team.team_barrier();
197 
198  const auto zt_grid_s = ekat::subview(zt_grid, i);
199  const auto zi_grid_s = ekat::subview(zi_grid, i);
200  const auto rrho_s = ekat::subview(rrho, i);
201  const auto rrho_i_s = ekat::subview(rrho_i, i);
202  SHF::linear_interp(team,zt_grid_s,zi_grid_s,rrho_s,rrho_i_s,nlev,nlev+1,0);
203  team.team_barrier();
204 
205  const auto exner_int = PF::exner_function(p_int(i,nlevi_v)[nlevi_p]);
206  const auto inv_exner_int_surf = 1/exner_int;
207 
208  wpthlp_sfc(i) = (surf_sens_flux(i)/(cpair*rrho_i(i,nlevi_v)[nlevi_p]))*inv_exner_int_surf;
209  wprtp_sfc(i) = surf_evap(i)/rrho_i(i,nlevi_v)[nlevi_p];
210  upwp_sfc(i) = surf_mom_flux(i,0)/rrho_i(i,nlevi_v)[nlevi_p];
211  vpwp_sfc(i) = surf_mom_flux(i,1)/rrho_i(i,nlevi_v)[nlevi_p];
212  } // operator

316  // Structure to handle the generation of data needed by the rest of the model based on output from
317  // shoc_main.
318  struct SHOCPostprocess {
319  SHOCPostprocess () = default;
320 
321  KOKKOS_INLINE_FUNCTION
322  void operator()(const Kokkos::TeamPolicy<KT::ExeSpace>::member_type& team) const
323  {
324  const int i = team.league_rank();
325 
326  const Real inv_qc_relvar_max = 10;
327  const Real inv_qc_relvar_min = 0.001;
328 
329  const int nlev_packs = ekat::npack<Spack>(nlev);
330  Kokkos::parallel_for(Kokkos::TeamVectorRange(team, nlev_packs), [&] (const Int& k)
331  {
332  // See comment in SHOCPreprocess::operator() about the necessity of *_copy views
333  tke(i,k) = tke_copy(i,k);
334  qc(i,k) = qc_copy(i,k);
335 
336  qv(i,k) = qw(i,k) - qc(i,k);
337 
338  cldfrac_liq(i,k) = ekat::min(cldfrac_liq(i,k), 1);
339 
340  //P3 uses inv_qc_relvar, P3 is using dry mmrs, but
341  //wet<->dry conversion is a constant factor that cancels out in mean(qc)^2/mean(qc'*qc').
342  inv_qc_relvar(i,k) = 1;
343  const auto condition = (qc(i,k) != 0 && qc2(i,k) != 0);
344  if (condition.any()) {
345  inv_qc_relvar(i,k).set(condition,
346  ekat::min(inv_qc_relvar_max,
347  ekat::max(inv_qc_relvar_min,
348  ekat::square(qc(i,k))/qc2(i,k))));
349  }
350 
351  // Temperature
352  const Spack dse_ik(dse(i,k));
353  const Spack z_mid_ik(z_mid(i,k));
354  const Real phis_i(phis(i));
355  T_mid(i,k) = PF::calculate_temperature_from_dse(dse_ik,z_mid_ik,phis_i);
356 
357  });
358 
359  // If necessary, set appropriate boundary fluxes for energy and mass conservation checks.
360  // Any boundary fluxes not included in SHOC interface are set to 0.
361  if (compute_mass_and_energy_fluxes) {
362  vapor_flux(i) = surf_evap(i);
363  water_flux(i) = 0.0;
364  ice_flux(i) = 0.0;
365  heat_flux(i) = surf_sens_flux(i);
366  }
367  } // operator
