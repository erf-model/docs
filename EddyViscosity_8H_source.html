137 InterpolateTurbulentViscosity (int i, int j, int k,
138  enum MomentumEqn momentumEqn,
139  enum DiffusionDir diffDir,
140  const amrex::Array4<const amrex::Real>& K_turb)
141 {
142  // Assuming we already have 'K_turb' computed for all (i, j, k)
143  amrex::Real turbViscInterpolated = 1.0;
144 
145  switch (momentumEqn) {
146  case MomentumEqn::x: // Reference face is x-face index (i, j, k)
147  switch (diffDir) {
148  case DiffusionDir::x:
149  turbViscInterpolated = K_turb(i-1, j, k, EddyDiff::Mom_h);
150  break;
151  case DiffusionDir::y:
152  turbViscInterpolated = 0.25*( K_turb(i-1, j , k, EddyDiff::Mom_h) + K_turb(i, j, k, EddyDiff::Mom_h)
153  + K_turb(i-1, j-1, k, EddyDiff::Mom_h) + K_turb(i, j-1, k, EddyDiff::Mom_h) );
154  break;
155  case DiffusionDir::z:
156  turbViscInterpolated = 0.25*( K_turb(i-1, j, k , EddyDiff::Mom_v) + K_turb(i, j, k, EddyDiff::Mom_v)
157  + K_turb(i-1, j, k-1, EddyDiff::Mom_v) + K_turb(i, j, k-1, EddyDiff::Mom_v) );
158  break;
159  }
160  break;
161  case MomentumEqn::y: // Reference face is y-face index (i, j, k)
162  switch (diffDir) {
163  case DiffusionDir::x:
164  turbViscInterpolated = 0.25*( K_turb(i , j-1, k, EddyDiff::Mom_h) + K_turb(i, j, k, EddyDiff::Mom_h)
165  + K_turb(i-1, j-1, k, EddyDiff::Mom_h) + K_turb(i-1, j, k, EddyDiff::Mom_h) );
166  break;
167  case DiffusionDir::y:
168  turbViscInterpolated = K_turb(i, j-1, k, EddyDiff::Mom_h);
169  break;
170  case DiffusionDir::z:
171  turbViscInterpolated = 0.25*( K_turb(i, j-1, k, EddyDiff::Mom_v) + K_turb(i, j, k, EddyDiff::Mom_v)
172  + K_turb(i, j-1, k-1, EddyDiff::Mom_v) + K_turb(i, j, k-1, EddyDiff::Mom_v) );
173  break;
174  }
175  break;
176  case MomentumEqn::z: // Reference face is z-face index (i, j, k)
177  switch (diffDir) {
178  case DiffusionDir::x:
179  turbViscInterpolated = 0.25*( K_turb(i , j, k-1, EddyDiff::Mom_v) + K_turb(i, j, k, EddyDiff::Mom_v)
180  + K_turb(i-1, j, k-1, EddyDiff::Mom_v) + K_turb(i-1, j, k, EddyDiff::Mom_v) );
181  break;
182  case DiffusionDir::y:
183  turbViscInterpolated = 0.25*( K_turb(i, j , k-1, EddyDiff::Mom_v) + K_turb(i, j, k, EddyDiff::Mom_v)
184  + K_turb(i, j-1, k-1, EddyDiff::Mom_v) + K_turb(i, j-1, k, EddyDiff::Mom_v) );
185  break;
186  case DiffusionDir::z:
187  turbViscInterpolated = K_turb(i, j, k-1, EddyDiff::Mom_v);
188  break;
189  }
190  break;
191  }
192 
193  return turbViscInterpolated;
