130 InterpolateTurbulentViscosity(const int &i, const int &j, const int &k,
131  const enum MomentumEqn &momentumEqn,
132  const enum DiffusionDir &diffDir,
133  const amrex::Array4<amrex::Real>& K_turb) {
134  // Assuming we already have 'K_turb' computed for all (i, j, k)
135  amrex::Real turbViscInterpolated = 1.0;
136 
137  switch (momentumEqn) {
138  case MomentumEqn::x: // Reference face is x-face index (i, j, k)
139  switch (diffDir) {
140  case DiffusionDir::x:
141  turbViscInterpolated = K_turb(i-1, j, k, EddyDiff::Mom_h);
142  break;
143  case DiffusionDir::y:
144  turbViscInterpolated = 0.25*( K_turb(i-1, j, k, EddyDiff::Mom_h) + K_turb(i, j, k, EddyDiff::Mom_h)
145  + K_turb(i-1, j-1, k, EddyDiff::Mom_h) + K_turb(i, j-1, k, EddyDiff::Mom_h) );
146  break;
147  case DiffusionDir::z:
148  turbViscInterpolated = 0.25*( K_turb(i-1, j, k, EddyDiff::Mom_v) + K_turb(i, j, k, EddyDiff::Mom_v)
149  + K_turb(i-1, j, k-1, EddyDiff::Mom_v) + K_turb(i, j, k-1, EddyDiff::Mom_v) );
150  break;
151  default:
152  amrex::Abort("Error: Diffusion direction is unrecognized");
153  }
154  break;
155  case MomentumEqn::y: // Reference face is y-face index (i, j, k)
156  switch (diffDir) {
157  case DiffusionDir::x:
158  turbViscInterpolated = 0.25*( K_turb(i, j-1, k, EddyDiff::Mom_h) + K_turb(i, j, k, EddyDiff::Mom_h)
159  + K_turb(i-1, j-1, k, EddyDiff::Mom_h) + K_turb(i-1, j, k, EddyDiff::Mom_h) );
160  break;
161  case DiffusionDir::y:
162  turbViscInterpolated = K_turb(i, j-1, k, EddyDiff::Mom_h);
163  break;
164  case DiffusionDir::z:
165  turbViscInterpolated = 0.25*( K_turb(i, j-1, k, EddyDiff::Mom_v) + K_turb(i, j, k, EddyDiff::Mom_v)
166  + K_turb(i, j-1, k-1, EddyDiff::Mom_v) + K_turb(i, j, k-1, EddyDiff::Mom_v) );
167  break;
168  default:
169  amrex::Abort("Error: Diffusion direction is unrecognized");
170  }
171  break;
172  case MomentumEqn::z: // Reference face is z-face index (i, j, k)
173  switch (diffDir) {
174  case DiffusionDir::x:
175  turbViscInterpolated = 0.25*( K_turb(i, j, k-1, EddyDiff::Mom_v) + K_turb(i, j, k, EddyDiff::Mom_v)
176  + K_turb(i-1, j, k-1, EddyDiff::Mom_v) + K_turb(i-1, j, k, EddyDiff::Mom_v) );
177  break;
178  case DiffusionDir::y:
179  turbViscInterpolated = 0.25*( K_turb(i, j, k-1, EddyDiff::Mom_v) + K_turb(i, j, k, EddyDiff::Mom_v)
180  + K_turb(i, j-1, k-1, EddyDiff::Mom_v) + K_turb(i, j-1, k, EddyDiff::Mom_v) );
181  break;
182  case DiffusionDir::z:
183  turbViscInterpolated = K_turb(i, j, k-1, EddyDiff::Mom_v);
184  break;
185  default:
186  amrex::Abort("Error: Diffusion direction is unrecognized");
187  }
188  break;
189  default:
190  amrex::Abort("Error: Momentum equation is unrecognized");
191  }
192 
193  return turbViscInterpolated;
