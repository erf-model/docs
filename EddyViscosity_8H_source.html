11 amrex::Real ComputeSmnSmn(const int& i,const int& j,const int& k,
12  const amrex::Array4<amrex::Real const>& u,
13  const amrex::Array4<amrex::Real const>& v,
14  const amrex::Array4<amrex::Real const>& w,
15  const amrex::GpuArray<amrex::Real, AMREX_SPACEDIM>& cellSizeInv,
16  const amrex::Box& domain, const amrex::BCRec* bc_ptr)
17 {
18  amrex::Real S11 = ComputeS11(i+1, j, k, u, cellSizeInv);
19  amrex::Real S22 = ComputeS22(i, j+1, k, v, cellSizeInv);
20  amrex::Real S33 = ComputeS33(i, j, k+1, w, cellSizeInv);
21 
22  amrex::Real S12 = 0.25* (
23  ComputeS12(i , j , k, u, v, w, cellSizeInv, domain, bc_ptr)
24  + ComputeS12(i , j+1, k, u, v, w, cellSizeInv, domain, bc_ptr)
25  + ComputeS12(i+1, j , k, u, v, w, cellSizeInv, domain, bc_ptr)
26  + ComputeS12(i+1, j+1, k, u, v, w, cellSizeInv, domain, bc_ptr)
27  );
28 
29  amrex::Real S13 = 0.25* (
30  ComputeS13(i , j, k , u, v, w, cellSizeInv, domain, bc_ptr)
31  + ComputeS13(i , j, k+1, u, v, w, cellSizeInv, domain, bc_ptr)
32  + ComputeS13(i+1, j, k , u, v, w, cellSizeInv, domain, bc_ptr)
33  + ComputeS13(i+1, j, k+1, u, v, w, cellSizeInv, domain, bc_ptr)
34  );
35 
36  amrex::Real S23 = 0.25* (
37  ComputeS23(i, j , k , u, v, w, cellSizeInv, domain, bc_ptr)
38  + ComputeS23(i, j , k+1, u, v, w, cellSizeInv, domain, bc_ptr)
39  + ComputeS23(i, j+1, k , u, v, w, cellSizeInv, domain, bc_ptr)
40  + ComputeS23(i, j+1, k+1, u, v, w, cellSizeInv, domain, bc_ptr)
41  );
42 
43  amrex::Real SmnSmn = S11*S11 + S22*S22 + S33*S33 + 2.0*S12*S12 + 2.0*S13*S13 + 2.0*S23*S23;
44 
45  return SmnSmn;
46 }

51 void ComputeTurbulentViscosity(const amrex::MultiFab& xvel, const amrex::MultiFab& yvel, const amrex::MultiFab& zvel,
52  const amrex::MultiFab& cons_in, amrex::MultiFab& eddyViscosity,
53  const amrex::Geometry& geom,
54  const SolverChoice& solverChoice,
55  const amrex::Gpu::DeviceVector<amrex::BCRec> domain_bcs_type_d)
56 {
57  const amrex::GpuArray<amrex::Real, AMREX_SPACEDIM> cellSizeInv = geom.InvCellSizeArray();
58 
59  const amrex::Real cellVol = 1.0 / (cellSizeInv[0] * cellSizeInv[1] * cellSizeInv[2]);
60  const amrex::Real Delta = std::pow(cellVol,1.0/3.0);
61 
62  const auto& domain = geom.Domain();
63  const auto& dom_lo = amrex::lbound(domain);
64  const auto& dom_hi = amrex::ubound(domain);
65 
66  // Compute the turbulent viscosity
67  if (solverChoice.les_type == LESType::Smagorinsky)
68  {
69  amrex::Real Cs = solverChoice.Cs;
70  amrex::Real CsDeltaSqr = Cs*Cs*Delta*Delta;
71 
72 #ifdef _OPENMP
73 #pragma omp parallel if (amrex::Gpu::notInLaunchRegion())
74 #endif
75  for ( amrex::MFIter mfi(eddyViscosity,amrex::TilingIfNotGPU()); mfi.isValid(); ++mfi) {
76 
77  const amrex::Box &bx = mfi.growntilebox(1);
78 
79  const amrex::Array4<amrex::Real const > &cell_data = cons_in.array(mfi);
80  const amrex::Array4<amrex::Real> &K = eddyViscosity.array(mfi);
81 
82  const amrex::Array4<amrex::Real const> &u = xvel.array(mfi);
83  const amrex::Array4<amrex::Real const> &v = yvel.array(mfi);
84  const amrex::Array4<amrex::Real const> &w = zvel.array(mfi);
85 
86  const amrex::BCRec* bc_ptr = domain_bcs_type_d.data();
87 
88  amrex::ParallelFor(bx, [=] AMREX_GPU_DEVICE (int i, int j, int k) noexcept
89  {
90  amrex::Real SmnSmn = ComputeSmnSmn(i,j,k,u,v,w,cellSizeInv,domain,bc_ptr);
91 
92  // Note the positive sign, which aligns well with the positive sign in the diffusion term for momentum equation
93  K(i, j, k, EddyDiff::Mom_h) = 2.0 * CsDeltaSqr * cell_data(i, j, k, Rho_comp) * std::sqrt(2.0*SmnSmn);
94  });
95  } //mfi
96  }
97  else if (solverChoice.les_type == LESType::Deardorff)
98  {
99  amrex::Real l_C_k = solverChoice.Ck;
100 
101 #ifdef _OPENMP
102 #pragma omp parallel if (amrex::Gpu::notInLaunchRegion())
103 #endif
104  for ( amrex::MFIter mfi(eddyViscosity,amrex::TilingIfNotGPU()); mfi.isValid(); ++mfi) {
105 
106  const amrex::Box &bx = mfi.growntilebox(1);
107 
108  const amrex::Array4<amrex::Real const > &cell_data = cons_in.array(mfi);
109  const amrex::Array4<amrex::Real> &K = eddyViscosity.array(mfi);
110 
111  amrex::ParallelFor(bx, [=] AMREX_GPU_DEVICE (int i, int j, int k) noexcept
112  {
113  // K = rho * C_k * Delta * KE^(1/2) = C_k * Delta * (rho * RhoKE)^1/2
114  K(i,j,k,EddyDiff::Mom_h) = l_C_k * Delta *
115  std::sqrt(cell_data(i,j,k,RhoKE_comp) * cell_data(i,j,k,Rho_comp));
116  });
117  } //mfi
118  }
119 
120  // Fill in the rest of the turbulent diffusivities
121  if (solverChoice.les_type == LESType::Smagorinsky ||
122  solverChoice.les_type == LESType::Deardorff) {
123  amrex::Real inv_Pr_t = solverChoice.Pr_t_inv;
124  amrex::Real inv_Sc_t = solverChoice.Sc_t_inv;
125  amrex::Real inv_sigma_k = 1.0 / solverChoice.sigma_k;
126  bool use_KE = (solverChoice.les_type == LESType::Deardorff);
127 
128 #ifdef _OPENMP
129 #pragma omp parallel if (amrex::Gpu::notInLaunchRegion())
130 #endif
131  for ( amrex::MFIter mfi(eddyViscosity,amrex::TilingIfNotGPU()); mfi.isValid(); ++mfi) {
132  const amrex::Box &bx = mfi.growntilebox(1);
133  const amrex::Array4<amrex::Real> &K = eddyViscosity.array(mfi);
134  amrex::ParallelFor(bx, [=] AMREX_GPU_DEVICE (int i, int j, int k) noexcept
135  {
136  // Get eddy diffusivities from the eddy viscosity
137  // Additional factor of 0.5 because K = 2 mu_t
138  K(i,j,k,EddyDiff::Theta_h) = 0.5 * K(i,j,k,EddyDiff::Mom_h) * inv_Pr_t;
139  K(i,j,k,EddyDiff::Scalar_h) = 0.5 * K(i,j,k,EddyDiff::Mom_h) * inv_Sc_t;
140  if (use_KE) {
141  K(i,j,k,EddyDiff::KE_h) = 0.5 * K(i,j,k,EddyDiff::Mom_h) * inv_sigma_k;
142  }
143  if (solverChoice.use_QKE && solverChoice.diffuse_QKE_3D) {
144  K(i,j,k,EddyDiff::QKE_h) = 0.5 * K(i,j,k,EddyDiff::Mom_h) * inv_sigma_k;
145  }
146 
147  // For LES: vertical and horizontal components are the same
148  K(i,j,k,EddyDiff::Mom_v) = K(i,j,k,EddyDiff::Mom_h);
149  K(i,j,k,EddyDiff::Theta_v) = K(i,j,k,EddyDiff::Theta_h);
150  K(i,j,k,EddyDiff::Scalar_v) = K(i,j,k,EddyDiff::Scalar_h);
151  if (use_KE) {
152  K(i,j,k,EddyDiff::KE_v) = K(i,j,k,EddyDiff::KE_h);
153  }
154  // QKE: vertical diffusion from PBL model
155  });
156  }
157  }
158 
159 #ifdef _OPENMP
160 #pragma omp parallel if (amrex::Gpu::notInLaunchRegion())
161 #endif
162  for ( amrex::MFIter mfi(eddyViscosity,amrex::TilingIfNotGPU()); mfi.isValid(); ++mfi) {
163 
164  const amrex::Box &bx = mfi.growntilebox(1);
165 
166  if (!(domain.contains(bx))) {
167  // Fill values outside the domain by straightforward extrapolation. Note this must be
168  // done separately from the loop above so all the interior values are filled. We also
169  // do this in three separate loops so that we don't have any race conditions.
170  const amrex::Array4<amrex::Real> &K = eddyViscosity.array(mfi);
171  amrex::ParallelFor(bx, (int) EddyDiff::NumDiffs, [=] AMREX_GPU_DEVICE (int i, int j, int k, int n) noexcept {
172  if (i < dom_lo.x) {
173  if (j < dom_lo.y)
174  K(i,j,k,n) = K(dom_lo.x,dom_lo.y,k,n);
175  else if (j > dom_hi.y)
176  K(i,j,k,n) = K(dom_lo.x,dom_hi.y,k,n);
177  else
178  K(i,j,k,n) = K(dom_lo.x,j,k,n);
179  } else if (i > dom_hi.x) {
180  if (j < dom_lo.y)
181  K(i,j,k,n) = K(dom_hi.x,dom_lo.y,k,n);
182  else if (j > dom_hi.y)
183  K(i,j,k,n) = K(dom_hi.x,dom_hi.y,k,n);
184  else
185  K(i,j,k,n) = K(dom_hi.x,j,k,n);
186  } else if (j < dom_lo.y) {
187  K(i,j,k,n) = K(i,dom_lo.y,k,n);
188  } else if (j > dom_hi.y) {
189  K(i,j,k,n) = K(i,dom_hi.y,k,n);
190  }
191  });
192  }
193  } //mfi
194 
195  // Fill interior ghost cells and any ghost cells outside a periodic domain
196  eddyViscosity.FillBoundary(geom.periodicity());
197 
198  // Now extend to low and high in vertical (assuming not periodic in z)
199  if (!geom.isPeriodic(2))
200  {
201 #ifdef _OPENMP
202 #pragma omp parallel if (amrex::Gpu::notInLaunchRegion())
203 #endif
204  for ( amrex::MFIter mfi(eddyViscosity,amrex::TilingIfNotGPU()); mfi.isValid(); ++mfi) {
205 
206  const amrex::Box &bx = mfi.growntilebox(1);
207 
208  if (!(domain.contains(bx))) {
209  const amrex::Array4<amrex::Real> &K = eddyViscosity.array(mfi);
210  amrex::ParallelFor(bx, (int) EddyDiff::NumDiffs, [=] AMREX_GPU_DEVICE (int i, int j, int k, int n) noexcept
211  {
212  if (k < dom_lo.z) K(i,j,k,n) = K(i,j,dom_lo.z,n);
213  if (k > dom_hi.z) K(i,j,k,n) = K(i,j,dom_hi.z,n);
214  });
215  }
216  } //mfi
217  }

239 inline amrex::Real
240 InterpolateTurbulentViscosity(const int &i, const int &j, const int &k,
241  const enum MomentumEqn &momentumEqn,
242  const enum DiffusionDir &diffDir,
243  const amrex::Array4<amrex::Real>& K_turb) {
244  // Assuming we already have 'K_turb' computed for all (i, j, k)
245  amrex::Real turbViscInterpolated = 1.0;
246 
247  switch (momentumEqn) {
248  case MomentumEqn::x: // Reference face is x-face index (i, j, k)
249  switch (diffDir) {
250  case DiffusionDir::x:
251  turbViscInterpolated = K_turb(i-1, j, k, EddyDiff::Mom_h);
252  break;
253  case DiffusionDir::y:
254  turbViscInterpolated = 0.25*( K_turb(i-1, j, k, EddyDiff::Mom_h) + K_turb(i, j, k, EddyDiff::Mom_h)
255  + K_turb(i-1, j-1, k, EddyDiff::Mom_h) + K_turb(i, j-1, k, EddyDiff::Mom_h) );
256  break;
257  case DiffusionDir::z:
258  turbViscInterpolated = 0.25*( K_turb(i-1, j, k, EddyDiff::Mom_v) + K_turb(i, j, k, EddyDiff::Mom_v)
259  + K_turb(i-1, j, k-1, EddyDiff::Mom_v) + K_turb(i, j, k-1, EddyDiff::Mom_v) );
260  break;
261  default:
262  amrex::Abort("Error: Diffusion direction is unrecognized");
263  }
264  break;
265  case MomentumEqn::y: // Reference face is y-face index (i, j, k)
266  switch (diffDir) {
267  case DiffusionDir::x:
268  turbViscInterpolated = 0.25*( K_turb(i, j-1, k, EddyDiff::Mom_h) + K_turb(i, j, k, EddyDiff::Mom_h)
269  + K_turb(i-1, j-1, k, EddyDiff::Mom_h) + K_turb(i-1, j, k, EddyDiff::Mom_h) );
270  break;
271  case DiffusionDir::y:
272  turbViscInterpolated = K_turb(i, j-1, k, EddyDiff::Mom_h);
273  break;
274  case DiffusionDir::z:
275  turbViscInterpolated = 0.25*( K_turb(i, j-1, k, EddyDiff::Mom_v) + K_turb(i, j, k, EddyDiff::Mom_v)
276  + K_turb(i, j-1, k-1, EddyDiff::Mom_v) + K_turb(i, j, k-1, EddyDiff::Mom_v) );
277  break;
278  default:
279  amrex::Abort("Error: Diffusion direction is unrecognized");
280  }
281  break;
282  case MomentumEqn::z: // Reference face is z-face index (i, j, k)
283  switch (diffDir) {
284  case DiffusionDir::x:
285  turbViscInterpolated = 0.25*( K_turb(i, j, k-1, EddyDiff::Mom_v) + K_turb(i, j, k, EddyDiff::Mom_v)
286  + K_turb(i-1, j, k-1, EddyDiff::Mom_v) + K_turb(i-1, j, k, EddyDiff::Mom_v) );
287  break;
288  case DiffusionDir::y:
289  turbViscInterpolated = 0.25*( K_turb(i, j, k-1, EddyDiff::Mom_v) + K_turb(i, j, k, EddyDiff::Mom_v)
290  + K_turb(i, j-1, k-1, EddyDiff::Mom_v) + K_turb(i, j-1, k, EddyDiff::Mom_v) );
291  break;
292  case DiffusionDir::z:
293  turbViscInterpolated = K_turb(i, j, k-1, EddyDiff::Mom_v);
294  break;
295  default:
296  amrex::Abort("Error: Diffusion direction is unrecognized");
297  }
298  break;
299  default:
300  amrex::Abort("Error: Momentum equation is unrecognized");
301  }
302 
303  return turbViscInterpolated;
304 }
