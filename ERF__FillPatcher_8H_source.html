125  constexpr amrex::Real eps = std::numeric_limits<float>::epsilon();
126  AMREX_ALWAYS_ASSERT((time >= m_crse_times[0]-eps) && (time <= m_crse_times[1]+eps));
127 
128  // Time interpolation factors
129  amrex::Real fac_new = (time - m_crse_times[0]) / m_dt_crse;
130  amrex::Real fac_old = 1.0 - fac_new;
131 
132  // Boundary condition operator
133  cbc(*(m_cf_crse_data_old), 0, m_ncomp, amrex::IntVect(0), time, 0);
134 
135  // Coarse MF to hold time interpolated data
136  amrex::MultiFab crse_data_time_interp(m_cf_crse_data_old->boxArray(),
137  m_cf_crse_data_old->DistributionMap(),
138  m_ncomp, amrex::IntVect{0});
139 
140  // Time interpolate the coarse data
141  amrex::MultiFab::LinComb(crse_data_time_interp,
142  fac_old, *(m_cf_crse_data_old), 0,
143  fac_new, *(m_cf_crse_data_new), 0,
144  0, m_ncomp, amrex::IntVect{0});
145 
146  // Call correct spatial interpolation type
147  amrex::IndexType m_ixt = mf.boxArray().ixType();
148  int ixt_sum = m_ixt[0]+m_ixt[1]+m_ixt[2];
149  if (ixt_sum == 0) {
150  InterpCell(mf,crse_data_time_interp,bcs,mask_val);
151  } else if (ixt_sum == 1) {
152  InterpFace(mf,crse_data_time_interp,mask_val);
153  } else {
