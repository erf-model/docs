243  // Perturbation amplitude calcluation
244  void calc_tpi_amp (const int boxIdx)
245  {
246  amrex::Real g = CONST_GRAV;
247  amrex::Real beta = 1./tpi_Tinfty;
248  amrex::Real Um = pb_mag[boxIdx];
249 
250  // Reset the perturbation amplitude
251  pb_amp[boxIdx] = 0.;
252 
253  #ifdef BULK_RICHARDSON_FORMULATION
254  // Ma and Senocak (2023) Eq. 17
255  pb_amp[boxIdx] = tpi_nonDim * Um * Um * Um / (g * beta * tpi_lref * tpi_Hpb);
256  #endif
257 
258  }

275  if (ubx.ok()) {
