129  // Perturbation update frequency check.
130  // This function will trigger calc_tpi_meanMag() and calc_tpi_amp(), when
131  // the local elapsed time is greater than the perturbation frequency
132  void calc_tpi_update (const int lev,
133  const amrex::Real dt,
134  amrex::MultiFab& mf_xvel,
135  amrex::MultiFab& mf_yvel,
136  amrex::MultiFab& mf_cons)
137  {
138  // Setting random number generator for update interval
139  srand( (unsigned) time(NULL) );
140 
141  auto m_ixtype = mf_cons.boxArray().ixType(); // safety step
142 
143  for (int boxIdx = 0; boxIdx < pb_ba[lev].size(); boxIdx++) {
144 
145  // Check if the local elapsed time is greater than the update interval
146  if ( pb_interval[boxIdx] <= pb_local_etime[boxIdx] ) {
147 
148  // Compute mean velocity magnitude within perturbation box
149  calc_tpi_meanMag_perBox(boxIdx, lev, mf_xvel, mf_yvel);
150  //calc_tpi_meanMag_firstCell(boxIdx, lev, mf_xvel, mf_yvel);
151 
152  // Done for parallelism to avoid Inf being stored in array
153  // Only the rank owning the box will be able to access the storage location
154  if (pb_mag[boxIdx] !=0.) {
155  amrex::Real interval = tpi_lref / pb_mag[boxIdx];
156  pb_interval[boxIdx] = RandomReal(0.9*interval,1.1*interval); // 10% variation
157  }
158 
159  // Trigger amplitude calculation per perturbation box
160  calc_tpi_amp(boxIdx, pb_interval[boxIdx]);
161 
162  // Trigger random amplitude storage per cell within perturbation box
163  // Random perturbation is applied as a white noise on the temperature source term
164  // it becomes colored through the filtering of temperature transport equation from
165  // the eddy diffusivity.
166  for (amrex::MFIter mfi(pb_cell,TileNoZ()); mfi.isValid(); ++mfi) {
167  amrex::Box vbx = mfi.validbox();
168  amrex::Box pbx = convert(pb_ba[lev][boxIdx], m_ixtype);
169  amrex::Box ubx = pbx & vbx;
170  if (ubx.ok()) {
171  amrex::Real amp_copy = pb_amp[boxIdx];
172  const amrex::Array4<amrex::Real> &pert_cell = pb_cell.array(mfi);
173  ParallelForRNG(ubx, [=] AMREX_GPU_DEVICE(int i, int j, int k, const amrex::RandomEngine& engine) noexcept {
174  amrex::Real rand_double = amrex::Random(engine);
175  pert_cell(i,j,k,0) = (rand_double*2.0 - 1.0) * amp_copy;
176  });
177  }
178  }
179 
180  // Reset local elapsed time
181  pb_local_etime[boxIdx] = 0.;
182  } else {
183  // Increase by timestep of level 0
184  pb_local_etime[boxIdx] += dt;
185  } // if
186  } // for
187  }

192  // Perturbation amplitude calculation
193  void calc_tpi_amp (const int boxIdx, const amrex::Real interval)
194  {
195  amrex::Real Um = pb_mag[boxIdx];
196  pb_amp[boxIdx] = 0.; // Safety step
197 
198  #ifdef ECKERT_FORMULATION
199  // Muñoz-Esparza et al. (2015) Eq. 9
200  pb_amp[boxIdx] = Um * Um / (tpi_rho * tpi_cp * tpi_nonDim);
201  #endif
202 
203  #ifdef BULK_RICHARDSON_FORMULATION
204  amrex::Real beta = 1./tpi_Tinf; // Thermal expansion coefficient
205 
206  // Pseudo Random temperature the ignores scale when mechanically tripping turbulence
207  amrex::Real g = (tpi_nonDim * Um * Um) / (tpi_Ti * tpi_Hpb);
208  //amrex::Real g = CONST_GRAV;
209 
210  // Ma and Senocak (2023) Eq. 8, solving for delta phi
211  if (g != 0) pb_amp[boxIdx] = (tpi_nonDim * Um * Um) / (g * beta * tpi_Hpb);
212  #endif
213 
214  // Ma & Senocak (2023) Eq. 7
215  pb_amp[boxIdx] /= interval;
216  }
