307  // Write/read the full NoahMP prognostic state (soil, snow incl. layers,
308  // canopy, aquifer, albedo history, ...) to/from a NetCDF restart directory
309  // so a restart reproduces a cold-start trajectory bitwise (issue #3255).
310  // Each local block writes its tile into the global-domain file collectively.
311  void
312  Write_Lsm_Restart (const std::string& dir) const override
313  {
314  for (const NoahmpIO_type& noahmpio : noahmpio_vect) {
315  const_cast<NoahmpIO_type&>(noahmpio).WriteRestart(dir);
316  }
317 
318  // Also save the precip snapshots (the cumulative precip Noah-MP last consumed).
319  // Restoring these on restart lets the next land call difference against the same
320  // baseline, so no precip is dropped or double-counted across the restart.
321  if (m_lev < int(m_precip_accum_prev.size())) {
322  for (int s(0); s < NoahmpPrecipSlot::NumSlots; ++s) {
323  if (m_precip_accum_prev[m_lev][s]) {
324  amrex::VisMF::Write(*m_precip_accum_prev[m_lev][s],
325  m_precip_snapshot_name(dir, m_lev, s));
326  }
327  }
328  }

338  // Read the saved precip snapshots into a staging store (the live snapshots are
339  // not allocated until the first Advance, which then copies these in). Legacy
340  // checkpoints have no such files -> store stays empty -> Advance cold-seeds.
