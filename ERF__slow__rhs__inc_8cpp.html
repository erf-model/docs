463  // Define updates and fluxes in the current RK stage
464  // *************************************************************************

484 
485  const Array4<const Real> & cell_data_old = S_old[IntVar::cons].array(mfi);

510  const Array4<const Real>& mf_u = mapfac_u->const_array(mfi);
511  const Array4<const Real>& mf_v = mapfac_v->const_array(mfi);

515  Array4<const Real> z_t;
516  if (z_t_mf)
517  z_t = z_t_mf->array(mfi);
518  else
519  z_t = Array4<const Real>{};

522  const Array4<Real>& rho_v_rhs = S_rhs[IntVar::ymom].array(mfi);
523  const Array4<Real>& rho_w_rhs = S_rhs[IntVar::zmom].array(mfi);
524 
525  const Array4<Real const>& mu_turb = l_use_turb ? eddyDiffs->const_array(mfi) : Array4<const Real>{};
526

528  const Array4<const Real>& z_nd = l_use_terrain ? z_phys_nd->const_array(mfi) : Array4<const Real>{};
529  const Array4<const Real>& detJ_arr = l_use_terrain ? detJ->const_array(mfi) : Array4<const Real>{};
530

532  const Array4<const Real>& p0_arr = p0->const_array(mfi);

535  // Contravariant flux field
536  //-----------------------------------------

538  BL_PROFILE("slow_rhs_making_omega");

540  // Now create Omega with momentum (not velocity)
541  amrex::ParallelFor(gbxo, [=] AMREX_GPU_DEVICE (int i, int j, int k) noexcept {
542  omega_arr(i,j,k) = rho_w(i,j,k);
543  });
544  } // end profile

549  //-----------------------------------------

551  Array4<Real> er_arr;

553  er_arr = expr->array(mfi);
554  } else {
555  er_arr = Array4<Real>{};

559  Array4<Real> tau11,tau22,tau33;
560  Array4<Real> tau12,tau13,tau23;

563  tau12 = Tau12->array(mfi); tau13 = Tau13->array(mfi); tau23 = Tau23->array(mfi);
564  } else {
565  tau11 = Array4<Real>{}; tau22 = Array4<Real>{}; tau33 = Array4<Real>{};
566  tau12 = Array4<Real>{}; tau13 = Array4<Real>{}; tau23 = Array4<Real>{};
567  }
568  // Terrain diffusion
569  Array4<Real> tau21,tau31,tau32;
570  if (Tau21) {
571  tau21 = Tau21->array(mfi); tau31 = Tau31->array(mfi); tau32 = Tau32->array(mfi);
572  } else {
573  tau21 = Array4<Real>{}; tau31 = Array4<Real>{}; tau32 = Array4<Real>{};
574  }
575 
576  // Strain magnitude
577  Array4<Real> SmnSmn_a;
578  if (solverChoice.les_type == LESType::Deardorff) {
579  SmnSmn_a = SmnSmn->array(mfi);
580  } else {
581  SmnSmn_a = Array4<Real>{};
582  }

585  // Define updates in the RHS of continuity, temperature, and scalar equations
586  // **************************************************************************
587  Real fac = 1.0;
588  AdvectionSrcForRhoAndTheta(bx, valid_bx, cell_rhs, // these are being used to build the fluxes
589  rho_u, rho_v, omega_arr, fac,
590  avg_xmom, avg_ymom, avg_zmom, // these are being defined from the rho fluxes
591  cell_prim, z_nd, detJ_arr,
592  dxInv, mf_m, mf_u, mf_v,
593  horiz_adv_type, vert_adv_type, l_use_terrain);
594 
595  if (l_use_diff) {
596  Array4<Real> diffflux_x = dflux_x->array(mfi);
597  Array4<Real> diffflux_y = dflux_y->array(mfi);
598  Array4<Real> diffflux_z = dflux_z->array(mfi);
599 
600  Array4<Real> hfx_z = Hfx3->array(mfi);
601  Array4<Real> diss = Diss->array(mfi);
602 
603  const Array4<const Real> tm_arr = t_mean_mf ? t_mean_mf->const_array(mfi) : Array4<const Real>{};
604 
605  // NOTE: No diffusion for continuity, so n starts at 1.
606  int n_start = amrex::max(start_comp,RhoTheta_comp);
607  int n_comp = end_comp - n_start + 1;
608 
609  DiffusionSrcForState_N(bx, domain, n_start, n_comp, u, v,
610  cell_data, cell_prim, cell_rhs,
611  diffflux_x, diffflux_y, diffflux_z,

614  mu_turb, solverChoice, tm_arr, grav_gpu, bc_ptr);

617  if (l_use_ndiff) {
618  NumericalDiffusion(bx, start_comp, num_comp, dt, solverChoice.NumDiffCoeff,
619  cell_data, cell_rhs, mf_u, mf_v, false, false);
620  }
621 
622  // Add source terms for (rho theta)
623  {
624  auto const& src_arr = source.const_array(mfi);
625  amrex::ParallelFor(bx, [=] AMREX_GPU_DEVICE (int i, int j, int k) noexcept
626  {
627  cell_rhs(i,j,k,RhoTheta_comp) += src_arr(i,j,k,RhoTheta_comp);
628  });
629  }
630 
631  // Add Rayleigh damping
632  if (solverChoice.use_rayleigh_damping && solverChoice.rayleigh_damp_T) {
633  int n = RhoTheta_comp;
634  int nr = Rho_comp;
635  int np = PrimTheta_comp;
636  amrex::ParallelFor(bx, [=] AMREX_GPU_DEVICE (int i, int j, int k) noexcept
637  {
638  Real theta = cell_prim(i,j,k,np);
639  cell_rhs(i, j, k, n) -= dptr_rayleigh_tau[k] * (theta - dptr_rayleigh_thetabar[k]) * cell_data(i,j,k,nr);
640  });
641  }
642 
643  // If in second RK stage, take average of old-time and new-time source
644  if (nrk == 1)
645  {
646  amrex::ParallelFor(bx, [=] AMREX_GPU_DEVICE (int i, int j, int k) noexcept
647  {
648  cell_rhs(i,j,k, Rho_comp) *= 0.5;
649  cell_rhs(i,j,k,RhoTheta_comp) *= 0.5;
650 
651  cell_rhs(i,j,k, Rho_comp) += 0.5 / dt * (cell_data(i,j,k, Rho_comp) - cell_data_old(i,j,k, Rho_comp));
652  cell_rhs(i,j,k,RhoTheta_comp) += 0.5 / dt * (cell_data(i,j,k,RhoTheta_comp) - cell_data_old(i,j,k,RhoTheta_comp));
653  });
654  }
655

657  // Define updates in the RHS of {x, y, z}-momentum equations
658  // *********************************************************************
659  AdvectionSrcForMom(tbx, tby, tbz,
660  rho_u_rhs, rho_v_rhs, rho_w_rhs, u, v, w,
661  rho_u , rho_v , omega_arr,
662  z_nd, detJ_arr, dxInv, mf_m, mf_u, mf_v,
663  horiz_adv_type, vert_adv_type, l_use_terrain, domhi_z);
664 
665  if (l_use_diff) {
666  DiffusionSrcForMom_N(tbx, tby, tbz,
667  rho_u_rhs, rho_v_rhs, rho_w_rhs,
668  tau11, tau22, tau33,
669  tau12, tau13, tau23,
670  cell_data, solverChoice, dxInv,
671  mf_m, mf_u, mf_v);
672  }
673 
674  if (l_use_ndiff) {
675  NumericalDiffusion(tbx, 0, 1, dt, solverChoice.NumDiffCoeff,
676  rho_u, rho_u_rhs, mf_m, mf_v, false, true);
677  NumericalDiffusion(tby, 0, 1, dt, solverChoice.NumDiffCoeff,
678  rho_v, rho_v_rhs, mf_u, mf_m, true, false);
679  NumericalDiffusion(tbz, 0, 1, dt, solverChoice.NumDiffCoeff,
680  rho_w, rho_w_rhs, mf_u, mf_v, false, false);
681  }
682 
683  {
684  BL_PROFILE("slow_rhs_inc_xmom");
685  // ******************************************************************
686  // NON-TERRAIN VERSION
687  // ******************************************************************
688  amrex::ParallelFor(tbx,
689  [=] AMREX_GPU_DEVICE (int i, int j, int k)
690  { // x-momentum equation
691 
692  // Note we do NOT include a pressure gradient here
693  rho_u_rhs(i, j, k) += - solverChoice.abl_pressure_grad[0]
694  + 0.5*(cell_data(i,j,k,Rho_comp)+cell_data(i-1,j,k,Rho_comp)) * solverChoice.abl_geo_forcing[0];
695 
696  // Add Coriolis forcing (that assumes east is +x, north is +y)
697  if (solverChoice.use_coriolis)
698  {
699  Real rho_v_loc = 0.25 * (rho_v(i,j+1,k) + rho_v(i,j,k) + rho_v(i-1,j+1,k) + rho_v(i-1,j,k));
700  Real rho_w_loc = 0.25 * (rho_w(i,j,k+1) + rho_w(i,j,k) + rho_w(i,j-1,k+1) + rho_w(i,j-1,k));
701  rho_u_rhs(i, j, k) += solverChoice.coriolis_factor *
702  (rho_v_loc * solverChoice.sinphi - rho_w_loc * solverChoice.cosphi);
703  }
704 
705  // Add Rayleigh damping
706  if (solverChoice.use_rayleigh_damping && solverChoice.rayleigh_damp_U)
707  {
708  Real uu = rho_u(i,j,k) / cell_data(i,j,k,Rho_comp);
709  rho_u_rhs(i, j, k) -= dptr_rayleigh_tau[k] * (uu - dptr_rayleigh_ubar[k]) * cell_data(i,j,k,Rho_comp);
710  }
711 
712  if (nrk == 1) {

715  }
716  });
717  } // end profile

726  { // y-momentum equation

729  rho_v_rhs(i, j, k) += - solverChoice.abl_pressure_grad[1]
730  + 0.5*(cell_data(i,j,k,Rho_comp)+cell_data(i,j-1,k,Rho_comp)) * solverChoice.abl_geo_forcing[1];
731 
732  // Add Coriolis forcing (that assumes east is +x, north is +y)
733  if (solverChoice.use_coriolis)
734  {
735  Real rho_u_loc = 0.25 * (rho_u(i+1,j,k) + rho_u(i,j,k) + rho_u(i+1,j-1,k) + rho_u(i,j-1,k));
736  rho_v_rhs(i, j, k) += -solverChoice.coriolis_factor * rho_u_loc * solverChoice.sinphi;
737  }
738 
739  // Add Rayleigh damping
740  if (solverChoice.use_rayleigh_damping && solverChoice.rayleigh_damp_V)
741  {
742  Real vv = rho_v(i,j,k) / cell_data(i,j,k,Rho_comp);
743  rho_v_rhs(i, j, k) -= dptr_rayleigh_tau[k] * (vv - dptr_rayleigh_vbar[k]) * cell_data(i,j,k,Rho_comp);
744  }
745 
746  if (nrk == 1) {
747  rho_v_rhs(i,j,k) *= 0.5;
748  rho_v_rhs(i,j,k) += 0.5 / dt * (rho_v(i,j,k) - rho_v_old(i,j,k));
749  }
750  });
751  } // end profile

754  BL_PROFILE("slow_rhs_inc_zmom_2d");

756  b2d.setSmall(2,0);
757  b2d.setBig(2,0);
758  // Enforce no forcing term at top and bottom boundaries
759  amrex::ParallelFor(b2d, [=] AMREX_GPU_DEVICE (int i, int j, int) {
760  rho_w_rhs(i,j, 0) = 0.;
761  rho_w_rhs(i,j,domhi_z+1) = 0.; // TODO: generalize this
762  });
763  } // end profile
764 
765  {
766  BL_PROFILE("slow_rhs_pre_zmom");
767  // ******************************************************************
768  // NON-TERRAIN VERSION
769  // ******************************************************************
770  amrex::ParallelFor(tbz,
771  [=] AMREX_GPU_DEVICE (int i, int j, int k)
772  { // z-momentum equation
773 
774  // Note we do NOT include a pressure gradient or buoyancy here
775  rho_w_rhs(i, j, k) += - solverChoice.abl_pressure_grad[2]
776  + 0.5*(cell_data(i,j,k,Rho_comp)+cell_data(i,j,k-1,Rho_comp)) * solverChoice.abl_geo_forcing[2];
777 
778  // Add Coriolis forcing (that assumes east is +x, north is +y)
779  if (solverChoice.use_coriolis)
780  {
781  Real rho_u_loc = 0.25 * (rho_u(i+1,j,k) + rho_u(i,j,k) + rho_u(i+1,j,k-1) + rho_u(i,j,k-1));
782  rho_w_rhs(i, j, k) += solverChoice.coriolis_factor * rho_u_loc * solverChoice.cosphi;
783  }
784 
785  // Add Rayleigh damping
786  if (solverChoice.use_rayleigh_damping && solverChoice.rayleigh_damp_W)
787  {
788  Real ww = rho_w(i,j,k) / cell_data(i,j,k,Rho_comp);
789  rho_w_rhs(i, j, k) -= dptr_rayleigh_tau[k] * (ww - dptr_rayleigh_wbar[k]) * cell_data(i,j,k,Rho_comp);
790  }
791 
792  if (nrk == 1) {
793  rho_w_rhs(i,j,k) *= 0.5;
794  rho_w_rhs(i,j,k) += 0.5 / dt * (rho_w(i,j,k) - rho_w_old(i,j,k));
795  }
796  });
797  } // end profile
798  } // mfi
