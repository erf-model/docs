114  inline
115  static void get_solar_band_fraction_irrad(real1d& fractional_irradiance) {
116  real tsi = 0; // total solar irradiance

145  // provide spectral boundaries of each longwave band
146  inline
147  static void get_lw_spectral_boundaries(real1d& low_boundaries,
148  real1d& high_boundaries,
149  Units units) {
150  switch (units) {
151  case inv_cm:
152  for(auto i = 0; i < nlwbands; ++i) {
153  low_boundaries(i) = wavenumber1_longwave[i];
154  high_boundaries(i) = wavenumber2_longwave[i];
155  }
156  case meter:
157  for(auto i = 0; i < nlwbands; ++i) {
158  low_boundaries(i) = 1.e-2/wavenumber2_longwave[i];
159  high_boundaries(i) = 1.e-2/wavenumber1_longwave[i];
160  }
161  case nanometer:
162  for(auto i = 0; i < nlwbands; ++i) {
163  low_boundaries(i) = 1.e7/wavenumber2_longwave[i];
164  high_boundaries(i) = 1.e7/wavenumber1_longwave[i];
165  }
166  case micrometer:
167  for(auto i = 0; i < nlwbands; ++i) {
168  low_boundaries(i) = 1.e4/wavenumber2_longwave[i];
169  high_boundaries(i) = 1.e4/wavenumber1_longwave[i];
170  }
171  case centimeter:
172  for(auto i = 0; i < nlwbands; ++i) {
173  low_boundaries(i) = 1./wavenumber2_longwave[i];
174  high_boundaries(i) = 1./wavenumber1_longwave[i];
175  }
176  default:
177  amrex::Print() << "get_lw_spectral_boundaries: spectral units not acceptable\n";
178  }
179  }
