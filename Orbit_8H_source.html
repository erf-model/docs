60  static constexpr amrex::Real day_p_yr = 365.0;
61  static constexpr amrex::Real ve = 80.5;
62 
63  // Compute eccentricity factor and solar declination using
64  // day value where a round day (such as 213.0) refers to 0z at
65  // Greenwich longitude.
66 
67  // Use formulas from Berger, Andre 1978: Long-Term Variations of Daily
68  // Insolation and Quaternary Climatic Changes. J. of the Atmo. Sci.
69  // 35:2362-2367.
70 
71  // To get the earths true longitude (position in orbit; lambda in Berger
72  // 1978) which is necessary to find the eccentricity factor and declination,
73  // must first calculate the mean longitude (lambda m in Berger 1978) at
74  // the present day. This is done by adding to lambm0 (the mean longitude
75  // at the vernal equinox, set as March 21 at noon, when lambda=0; in radians)
76  // an increment (delta lambda m in Berger 1978) that is the number of
77  // days past or before (a negative increment) the vernal equinox divided by
78  // the days in a model year times the 2*pi radians in a complete orbit.
79  amrex::Real lambm = lambm0 + (calday - ve)*2.0*PI/day_p_yr;
80  amrex::Real lmm = lambm - mvelpp;
81 
82  // The earths true longitude, in radians, is then found from
83  // the formula in Berger 1978:
84  amrex::Real sinl = std::sin(lmm);
85  amrex::Real lamb = lambm + eccen*(2.0*sinl + eccen*(1.25*sin(2.0*lmm)
86  + eccen*((13.0/12.0)*sin(3.0*lmm) - 0.25*sinl)));
87 
88 
89  // Using the obliquity, eccentricity, moving vernal equinox longitude of
90  // perihelion (plus), and earths true longitude, the declination (delta)
91  // and the normalized earth/sun distance (rho in Berger 1978; actually inverse
92  // rho will be used), and thus the eccentricity factor (eccf), can be
93  // calculated from formulas given in Berger 1978.
94  amrex::Real invrho = (1.0 + eccen*std::cos(lamb - mvelpp)) / (1.0 - eccen*eccen);
95 
96  // Set solar declination and eccentricity factor
97  delta = std::asin(std::sin(obliqr)*std::sin(lamb));
98  eccf = invrho*invrho;
99 }
