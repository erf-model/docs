115  AMREX_GPU_HOST_DEVICE AMREX_FORCE_INLINE
116  amrex::Real calc_psi_m(amrex::Real zeta) const
117  {
118  if (zeta > 0) {
119  return -beta_m * zeta;
120  } else {
121  amrex::Real x = std::sqrt(std::sqrt(1 - gamma_m * zeta));
122  return 2.0 * std::log(0.5 * (1.0 + x)) + log(0.5 * (1 + x * x)) -
123  2.0 * std::atan(x) + 1.57;
124  }
125  }
