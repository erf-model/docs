71  AMREX_GPU_HOST_DEVICE AMREX_FORCE_INLINE
72  amrex::Real calc_psi_m(amrex::Real zeta) const
73  {
74  if (zeta > 0) {
75  return -beta_m * zeta;
76  } else {
77  amrex::Real x = std::sqrt(std::sqrt(1 - gamma_m * zeta));
78  return 2.0 * std::log(0.5 * (1.0 + x)) + log(0.5 * (1 + x * x)) -
79  2.0 * std::atan(x) + 1.57;
80  }
81  }

87  return -beta_h * zeta;
88  } else {
89  amrex::Real x = std::sqrt(1 - gamma_h * zeta);
