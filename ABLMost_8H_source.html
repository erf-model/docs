72  AMREX_GPU_HOST_DEVICE AMREX_FORCE_INLINE
73  amrex::Real calc_psi_m(amrex::Real zeta) const
74  {
75  if (zeta > 0) {
76  return -gamma_m * zeta;
77  } else {
78  amrex::Real x = std::sqrt(std::sqrt(1 - beta_m * zeta));
79  return 2.0 * std::log(0.5 * (1.0 + x)) + log(0.5 * (1 + x * x)) -

98 inline void ABLMost::update_fluxes(int max_iters)
99 {
100  constexpr amrex::Real eps = 1.0e-16;
101  amrex::Real zeta = 0.0;
102  amrex::Real utau_iter = 0.0;
103 
104  // Initialize variables
105  amrex::Real psi_m = 0.0;
106  amrex::Real psi_h = 0.0;
107  utau = kappa * vmag_mean / (std::log(zref / z0));
108 
109  int iter = 0;
110  do {
111  utau_iter = utau;
112  switch (alg_type) {
113  case HEAT_FLUX:
114  surf_temp = surf_temp_flux * (std::log(zref / z0) - psi_h) /
115  (utau * kappa) + theta_mean;
116  break;
117 
118  case SURFACE_TEMPERATURE:
119  surf_temp_flux = -(theta_mean - surf_temp) * utau * kappa /
120  (std::log(zref / z0) - psi_h);
121  break;
122  }
123 
124  if (std::abs(surf_temp_flux) > eps) {
125  // Stable and unstable ABL conditions
126  obukhov_len = -utau * utau * utau * theta_mean /
127  (kappa * gravity * surf_temp_flux);
128  zeta = zref / obukhov_len;
129  } else {
130  // Neutral conditions
131  obukhov_len = std::numeric_limits<amrex::Real>::max();
132  zeta = 0.0;
133  }
134  psi_m = calc_psi_m(zeta);
135  psi_h = calc_psi_h(zeta);
136  utau = kappa * vmag_mean / (std::log(zref / z0) - psi_m);
137  ++iter;
138  } while ((std::abs(utau_iter - utau) > 1e-5) && iter <= max_iters);
139 
140  if (iter >= max_iters) {
141  amrex::Print()
142  << "MOData::update_fluxes: Convergence criteria not met after "
143  << max_iters << " iterations"
144  << "\nObuhov length = " << obukhov_len << " zeta = " << zeta
145  << "\npsi_m = " << psi_m << " psi_h = " << psi_h
146  << "\nutau = " << utau << " Tsurf = " << surf_temp
147  << " q = " << surf_temp_flux << std::endl;
148  }
149 }
