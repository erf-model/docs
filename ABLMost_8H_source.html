40  AMREX_GPU_HOST_DEVICE AMREX_FORCE_INLINE
41  amrex::Real calc_psi_m(amrex::Real zeta) const
42  {
43  if (zeta > 0) {
44  return -beta_m * zeta;
45  } else {
46  amrex::Real x = std::sqrt(std::sqrt(1.0 - gamma_m * zeta));
47  return 2.0 * std::log(0.5 * (1.0 + x)) + log(0.5 * (1.0 + x * x)) -
48  2.0 * std::atan(x) + PIoTwo;
49  }
50  }
51 
52  AMREX_GPU_HOST_DEVICE AMREX_FORCE_INLINE
53  amrex::Real calc_psi_h(amrex::Real zeta) const
54  {
55  if (zeta > 0) {
56  return -beta_h * zeta;
57  } else {
58  amrex::Real x = std::sqrt(1.0 - gamma_h * zeta);
59  return 2.0 * std::log(0.5 * (1.0 + x));
60  }
61  }
