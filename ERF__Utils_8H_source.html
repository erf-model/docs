497 AMREX_GPU_HOST_DEVICE
498 AMREX_FORCE_INLINE
499 amrex::Real
500 quad_interp_1d(amrex::Real z,

502  amrex::Real z1, amrex::Real p1,
503  amrex::Real z2, amrex::Real p2)
