73 AMREX_FORCE_INLINE
74 amrex::Real
75 MOSTAverage::line_average_interpolated(amrex::Real x, int icomp, int imf) const
76 {
77  int ncomp = m_ncomps[imf];
78  AMREX_ALWAYS_ASSERT(icomp >= 0 && icomp < ncomp);
79 
80  amrex::Real c = 0.0;
81  int ind = 0;
82 
83  if (x > m_xlo + 0.5 * m_dx) {
84  ind = static_cast<int>(floor((x - m_xlo) / m_dx - 0.5));
85  const amrex::Real x1 = m_xlo + (ind + 0.5) * m_dx;
86  c = (x - x1) / m_dx;
87  }
88 
89  if (ind + 1 >= m_ncell_line) {
90  ind = m_ncell_line - 2;
91  c = 1.0;
92  }
93 
94  AMREX_ALWAYS_ASSERT(ind >= 0 && ind + 1 < m_ncell_line);
95 
96  return m_line_average[imf][ncomp * ind + icomp] * (1.0 - c) +
97  m_line_average[imf][ncomp * (ind + 1) + icomp] * c;
98 }
