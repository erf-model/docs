83 AMREX_FORCE_INLINE
84 void
85 MOSTAverage::line_average_interpolated(amrex::Real x, int icomp, int imf) const
86 {
87  int ncomp = m_ncomps[imf];
88  AMREX_ALWAYS_ASSERT(icomp >= 0 && icomp < ncomp);
89 
90  amrex::Real c = 0.0;
91  int ind = 0;
92 
93  if (x > m_xlo + 0.5 * m_dx) {
94  ind = static_cast<int>(floor((x - m_xlo) / m_dx - 0.5));
95  const amrex::Real x1 = m_xlo + (ind + 0.5) * m_dx;
96  c = (x - x1) / m_dx;
97  }
98 
99  if (ind + 1 >= m_ncell_line) {
100  ind = m_ncell_line - 2;
101  c = 1.0;
102  }
103 
104  AMREX_ALWAYS_ASSERT(ind >= 0 && ind + 1 < m_ncell_line);
105 
106  // Interpolated value between planar averages
107  amrex::Real avg = m_line_average[imf][ncomp * ind + icomp] * (1.0 - c) +
108  m_line_average[imf][ncomp * (ind + 1) + icomp] * c;
109 
110  // Set the 2D multifab to a constant value
111  m_averages[imf]->setVal(avg);
112 }
