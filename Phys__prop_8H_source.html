87  void physprop_accum_unique_files(const std::string& filename,
88  const std::string& type)
89  {
90  // Count number of aerosols in input radname array. Aerosols are identified
91  // as strings with a ".nc" suffix.
92  // Construct a cumulative list of unique filenames containing physical property data.
93 
94  // check if filename is either a bulk aerosol or a mode
95  if (type == "A" || type == "M") {
96  // check if this filename has been used by another aerosol. If not
97  // then add it to the list of unique names.
98  if (physprop_get_id(filename) < 0)
99  uniquefilenames.push_back(filename);
100  }
101  }

103  // Read properties from the aerosol data files.
104  // ***N.B.*** The calls to physprop_accum_unique_files must be made before calling
105  // this init routine. physprop_accum_unique_files is responsible for building
106  // the list of files to be read here.
107  void physprop_init()
108  {
109  int numphysprops = uniquefilenames.size();
110  physprop.reserve(numphysprops);
111 
112  for(auto fileindex = 0; fileindex < numphysprops; ++fileindex) {
113  physprop[fileindex].sourcefile = uniquefilenames[fileindex];
114  aerosol_optics_init(physprop[fileindex]);
115  }
116  }

118  // Look for filename in the global list of unique filenames (module data uniquefilenames).
119  // If found, return it's index in the list. Otherwise return -1.
120  int physprop_get_id(std::string filename)
121  {
122  auto physprop_get_id = -1;
123  auto numphysprops = uniquefilenames.size();
124  for(auto iphysprop = 0; iphysprop < numphysprops; ++iphysprop) {
125  if(uniquefilenames[iphysprop] == filename) {
126  physprop_get_id = iphysprop;
127  return 0;
128  }
129  }
130  return physprop_get_id;
131  }

397  // Determine the opticstype, then call the
398  // appropriate routine to read the data.
399  void aerosol_optics_init(physprop_t& phys_prop)
400  {
401  using charHost1d = FArray<char,1,yakl::memHost>;
402  yakl::SimpleNetCDF prop;
403  prop.open(phys_prop.sourcefile, yakl::NETCDF_MODE_READ);
404 
405  charHost1d temp;
406  prop.read(temp, "opticsmethod");
407  phys_prop.opticsmethod = "";
408  for (auto ichar = 0 ; ichar <= temp.extent(0); ichar++)
409  if (!isspace(temp(ichar))) phys_prop.opticsmethod += temp(ichar);
410 
411  if(strcmp(phys_prop.opticsmethod.c_str(),"zero") == 0) {
412  zero_optics_init(phys_prop, prop);
413  } else if (strcmp(phys_prop.opticsmethod.c_str(),"hygro") == 0) {
414  hygro_optics_init(phys_prop, prop);
415  } else if (strcmp(phys_prop.opticsmethod.c_str(),"hygroscopic") == 0) {
416  hygroscopic_optics_init(phys_prop, prop);
417  } else if (strcmp(phys_prop.opticsmethod.c_str(),"nonhygro") == 0) {
418  nonhygro_optics_init(phys_prop, prop);
419  } else if (strcmp(phys_prop.opticsmethod.c_str(),"insoluble") == 0) {
420  insoluble_optics_init(phys_prop, prop);
421  } else if (strcmp(phys_prop.opticsmethod.c_str(),"volcanic_radius") == 0) {
422  volcanic_radius_optics_init(phys_prop, prop);
423  } else if (strcmp(phys_prop.opticsmethod.c_str(),"volcanic") == 0) {
424  volcanic_optics_init(phys_prop, prop);
425  } else if (strcmp(phys_prop.opticsmethod.c_str(),"modal") == 0) {
426  modal_optics_init(phys_prop, prop);
427  } else {
428  amrex::Print() << "no options available\n";
429  }
430  }

432  void hygro_optics_init(physprop_t& phys_prop, yakl::SimpleNetCDF& prop)
433  {
434  real1d frh;
435  real2d fsw_ext;
436  real2d fsw_ssa;
437  real2d fsw_asm;
438  real2d flw_abs;
439 
440  auto nrh = RadConstants::nrh;
441  auto nbnd = prop.getDimSize( "lw_band" );
442  auto nswbands = prop.getDimSize( "sw_band" );
443 
444  prop.read( fsw_ext, "ext_sw");
445  prop.read( fsw_ssa, "ssa_sw");
446  prop.read( fsw_asm, "asm_sw");
447  prop.read( flw_abs, "abs_lw");
448  prop.read( frh, "rh");
449 
450  real1d fswe("fswe",nrh),
451  fsws("fsws",nrh),
452  fswa("fswa",nrh);
453 
454  // interpolate onto cam's rh mesh
455  for(auto kbnd = 0; kbnd < nswbands; ++kbnd) {
456  for(auto krh = 0; krh < nrh; ++krh) {
457  fswe(krh) = fsw_ext(krh, kbnd)/fsw_ext(1, kbnd);
458  fsws(krh) = fsw_ssa(krh, kbnd)/fsw_ssa(1, kbnd);
459  fswa(krh) = fsw_asm(krh, kbnd)/fsw_asm(1, kbnd);
460  }
461 
462  // interpolation
463  for(auto krh = 0; krh < nrh; ++krh) {
464  auto rh = 1.0 / nrh * (krh - 1);
465  phys_prop.sw_hygro_ext(krh, kbnd) =
466  exp_interpol( frh, fswe, rh ) * fsw_ext(1, kbnd);
467  phys_prop.sw_hygro_ssa(krh,kbnd) =
468  lin_interpol( frh, fsws, rh ) * fsw_ssa(1, kbnd);
469  phys_prop.sw_hygro_asm(krh,kbnd) =
470  lin_interpol( frh, fswa, rh ) * fsw_asm(1, kbnd);
471  }
472  }
473 
474  // read refractive index data if available
475  refindex_aer_init(phys_prop, prop);
476 
477  // read bulk aero props
478  bulk_props_init(phys_prop, prop);
479  }

689  // Purpose:
690  //interpolates f(x) to point y
691  //assuming f(x) = f(x0) exp a(x - x0)
692  //where a = ( ln f(x1) - ln f(x0) ) / (x1 - x0)
693  //x0 <= x <= x1
694  //assumes x is monotonically increasing
695  real exp_interpol(const real1d& x, const real1d& f, const real& y)
696  {
697  auto n = x.extent(0);
698  int k;
699 
700  // find k such that x(k) < y =< x(k+1)
701  // set k = 1 if y <= x(1) and k = n-1 if y > x(n)
702  if (y <= x(1)) {
703  k = 1;
704  }
705  else if (y >= x(n)) {
706  k = n - 1;
707  }
708  else {
709  k = 1;
710  while (y > x(k+1) && k < n) {
711  k = k + 1;
712  }
713  }
714 
715  // interpolate
716  auto a = ( log( f(k+1) / f(k) ) ) / ( x(k+1) - x(k) );
717  return f(k) * exp( a * (y - x(k)) );
718  }
719 
720  //Purpose:
721  // interpolates f(x) to point y
722  // assuming f(x) = f(x0) + a * (x - x0)
723  // where a = ( f(x1) - f(x0) ) / (x1 - x0)
724  // x0 <= x <= x1
725  // assumes x is monotonically increasing
726  //
727  real lin_interpol(const real1d& x, const real1d& f, const real& y)
728  {
729  auto n = x.extent(0);
730  int k;
731  // find k such that x(k) < y =< x(k+1)
732  // set k = 1 if y <= x(1) and k = n-1 if y > x(n)
733  if (y <= x(1)) {
734  k = 1;
735  }
736  else if (y >= x(n)) {
737  k = n - 1;
738  }
739  else {
740  k = 1;
741  while (y > x(k+1) && k < n) {
742  k = k + 1;
743  }
744  }
745  // interpolate
746  auto a = (f(k+1)-f(k))/(x(k+1)-x(k));
747  return f(k)+a*(y-x(k));
748  }
