525  AMREX_FORCE_INLINE
526  int
527  ComputeGhostCells(const int& spatial_order) {
528  int nGhostCells;
529 
530  switch (spatial_order) {
531  case 2:
532  nGhostCells = 2; // We need this many to compute the eddy viscosity in the ghost cells
533  break;
534  case 3:
535  nGhostCells = 2;
536  break;
537  case 4:
538  nGhostCells = 2;
539  break;
540  case 5:
541  nGhostCells = 3;
542  break;
543  case 6:
544  nGhostCells = 3;
545  break;
546  default:
547  amrex::Error("Must specify spatial order to be 2,3,4,5 or 6");
548  }
549 
550  return nGhostCells;
551  }
