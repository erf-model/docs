144  const amrex::Real rho_qps,

146 {
147  // Copy-in clips moisture and stores pressure in mbar.

234 SAMCloudPhaseChange sam_partition_cloud_phase (const int SAM_moisture_type,
235  const amrex::Real tabs,
236  const amrex::Real qn,
237  const amrex::Real qcl,
238  const amrex::Real qci,
239  const amrex::Real fac_cond,
240  const amrex::Real fac_fus,
241  const amrex::Real an,
242  const amrex::Real bn) noexcept
243 {
244  SAMCloudPhaseChange result{};
245  result.qcl = qcl;
246  result.qci = qci;
247  result.tabs = tabs;
248  result.omn = sam_cloud_liquid_fraction(SAM_moisture_type, tabs, an, bn);
249 
250  if (SAM_moisture_type == 1) {
251  if (tabs >= tbgmax) {
252  result.omn = one;
253  result.delta_qi = qci;
254  result.qci = zero;

259  result.delta_qc = qcl;
260  result.qcl = zero;
261  result.qci += result.delta_qc;
262  result.tabs += fac_fus * result.delta_qc;
263  } else {
264  result.delta_qc = qcl - qn * result.omn;
265  result.delta_qi = qci - qn * (one - result.omn);
266  result.qcl = qn * result.omn;
267  result.qci = qn * (one - result.omn);
268  result.tabs += fac_fus * result.delta_qc;
269  }
270  } else {
271  result.omn = one;
272  result.delta_qc = qcl - qn;
273  result.qcl = qn;
274  result.qci = zero;
275  result.tabs += fac_cond * result.delta_qc;
276  }
277 
278  result.qn = result.qcl + result.qci;
279  return result;

471 AMREX_GPU_HOST_DEVICE AMREX_FORCE_INLINE
472 SAMCoefficientRow sam_compute_coefficient_row (const amrex::Real rho,
473  const amrex::Real tabs,
474  const amrex::Real gamr1,
475  const amrex::Real gamr2,
476  const amrex::Real gams1,
477  const amrex::Real gams2,
478  const amrex::Real gamg1,
479  const amrex::Real gamg2) noexcept
480 {
481  // Preserve the SAM baseline formula ordering in these closure rows.
482  SAMCoefficientRow row{};
483  amrex::Real prefactor;
484  amrex::Real pratio = std::sqrt(amrex::Real(1.29) / rho);
485  amrex::Real estw = amrex::Real(100.0) * erf_esatw(tabs);
486  amrex::Real esti = amrex::Real(100.0) * erf_esati(tabs);
487 
488  amrex::Real coef1 = fourth * PI * nzeros * a_snow * gams1 * pratio
489  / std::pow((PI * rhos * nzeros / rho), ((three + b_snow) / amrex::Real(4.0)));
490  amrex::Real coef2 = std::exp(amrex::Real(0.025) * (tabs - amrex::Real(273.15)));
491  row.accrsi = coef1 * coef2 * esicoef;
492  row.accrsc = coef1 * esccoef;
493  row.coefice = coef2;
494 
495  coef1 = (lsub / (tabs * R_v) - one) * lsub / (therco * tabs);
496  coef2 = R_v * R_d / (diffelq * esti);
497  prefactor = two * PI * nzeros / (rho * (coef1 + coef2));
498  prefactor *= (two / PI);
499  row.evaps1 = prefactor * amrex::Real(0.65) * std::sqrt(rho / (PI * rhos * nzeros));
500  row.evaps2 = prefactor * amrex::Real(0.44) * std::sqrt(a_snow * rho / muelq) * gams2
501  * std::sqrt(pratio) * std::pow(rho / (PI * rhos * nzeros), ((amrex::Real(5.0) + b_snow) / amrex::Real(8.0)));
502 
503  coef1 = fourth * PI * nzerog * a_grau * gamg1 * pratio
504  / std::pow((PI * rhog * nzerog / rho), ((three + b_grau) / amrex::Real(4.0)));
505  coef2 = std::exp(amrex::Real(0.025) * (tabs - amrex::Real(273.15)));
506  row.accrgi = coef1 * coef2 * egicoef;
507  row.accrgc = coef1 * egccoef;
508 
509  coef1 = (lsub / (tabs * R_v) - one) * lsub / (therco * tabs);
510  coef2 = R_v * R_d / (diffelq * esti);
511  prefactor = two * PI * nzerog / (rho * (coef1 + coef2));
512  row.evapg1 = prefactor * amrex::Real(0.78) * std::sqrt(rho / (PI * rhog * nzerog));
513  row.evapg2 = prefactor * amrex::Real(0.31) * std::sqrt(a_grau * rho / muelq) * gamg2
514  * std::sqrt(pratio) * std::pow(rho / (PI * rhog * nzerog), ((amrex::Real(5.0) + b_grau) / amrex::Real(8.0)));
515 
516  row.accrrc = fourth * PI * nzeror * a_rain * gamr1 * pratio
517  / std::pow((PI * rhor * nzeror / rho), ((three + b_rain) / amrex::Real(4.0))) * erccoef;
518 
519  coef1 = (lcond / (tabs * R_v) - one) * lcond / (therco * tabs);
520  coef2 = R_v * R_d / (diffelq * estw);
521  prefactor = two * PI * nzeror / (rho * (coef1 + coef2));

579 AMREX_GPU_HOST_DEVICE AMREX_FORCE_INLINE
580 SAMPrecipFaceState sam_precip_face_state (
581  const int SAM_moisture_type,
582  const amrex::Array4<const amrex::Real>& rho_array,
583  const amrex::Array4<const amrex::Real>& tabs_array,
584  const amrex::Array4<const amrex::Real>& qp_array,
585  const int i,
586  const int j,
587  const int k,
588  const int k_lo,
589  const int k_hi) noexcept
590 {
591  // MUST MATCH: SAM PrecipFall bottom/interior/top face rules.
592  SAMPrecipFaceState result{};
593  if (k == k_lo) {
594  result.rho_avg = rho_array(i,j,k);
595  result.tabs_avg = tabs_array(i,j,k);
596  result.qp_avg = qp_array(i,j,k);
597  } else if (k == k_hi + 1) {
598  result.rho_avg = rho_array(i,j,k-1);
599  result.tabs_avg = tabs_array(i,j,k-1);
600  result.qp_avg = qp_array(i,j,k-1);
601  } else {
602  result.rho_avg = myhalf * (rho_array(i,j,k-1) + rho_array(i,j,k));
603  result.tabs_avg = myhalf * (tabs_array(i,j,k-1) + tabs_array(i,j,k));
604  result.qp_avg = myhalf * (qp_array(i,j,k-1) + qp_array(i,j,k));
605  }
606  result.omp = sam_precip_rain_fraction(SAM_moisture_type, result.tabs_avg);
607  result.omg = sam_graupel_fraction(SAM_moisture_type, result.tabs_avg);
608  result.qrr = result.omp * result.qp_avg;
609  result.qss = (one - result.omp) * (one - result.omg) * result.qp_avg;
610  result.qgg = (one - result.omp) * result.omg * result.qp_avg;
611  return result;
612 }

615 amrex::Real sam_precip_flux_from_face_state (const SAMPrecipFaceState& face_state,
616  const amrex::Real vrain,
617  const amrex::Real vsnow,
618  const amrex::Real vgrau) noexcept
619 {
620  if (face_state.qp_avg <= qp_threshold) {

625  * ((one - face_state.omg) * vsnow * std::pow(face_state.rho_avg * face_state.qss, one + csnow)
626  + face_state.omg * vgrau * std::pow(face_state.rho_avg * face_state.qgg, one + cgrau));
627 }
