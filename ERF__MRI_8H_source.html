133  amrex::Real advance (T& S_old, T& S_new, amrex::Real time, const amrex::Real time_step)
134  {
135  BL_PROFILE_REGION("MRI_advance");
136  using namespace amrex;
137 
138  // *******************************************************************************
139  // version == 0: we only update the fast variables every fast timestep, then update
140  // the slow variables after the acoustic sub-stepping. This has
141  // two calls to slow_rhs so that we can update the slow variables
142  // with the velocity field after the acoustic substepping using
143  // the time-averaged velocity from the substepping
144  // version == 1: we don't do any acoustic subcyling so we only make one call per RK
145  // stage to slow_rhs
146  // *******************************************************************************
147  int version = no_substepping ? 1 : 0;
148 
149  timestep = time_step;
150 
151  const int substep_ratio = get_slow_fast_timestep_ratio();
152 
153  AMREX_ALWAYS_ASSERT(substep_ratio > 1 && substep_ratio % 2 == 0);
154 
155  const amrex::Real sub_timestep = timestep / substep_ratio;
156 
157  // Assume before advance() that S_old is valid data at the current time ("time" argument)
158  // And that if data is a MultiFab, both S_old and S_new contain ghost cells for evaluating a stencil based RHS
159  // We need this from S_old. This is convenient for S_new to have so we can use it
160  // as scratch space for stage values without creating a new scratch MultiFab with ghost cells.
161 
162  // NOTE: In the following, we use S_new to hold S*, S**, and finally, S^(n+1) at the new time
163  // DEFINITIONS:
164  // S_old = S^n
165  // S_sum = S(t)
166  // F_slow = F(S_stage)
167  // F_pert = G(S(t)-S_stage, S_stage)
168 
169  int n_data = IntVar::NumVars;
170 
171  /**********************************************/
172  /* RK3 Integration with Acoustic Sub-stepping */
173  /**********************************************/
174 
175  // Start with S_new (aka S_stage) holding S_old
176  #ifdef _OPENMP
177  #pragma omp parallel if (amrex::Gpu::notInLaunchRegion())
178  #endif
179  {
180  for ( MFIter mfi(S_old[IntVar::cons],TilingIfNotGPU()); mfi.isValid(); ++mfi) {
181  const Box gbx = mfi.tilebox().grow(S_old[IntVar::cons].nGrowVect());
182  const Box gtbx = mfi.nodaltilebox(0).grow(S_old[IntVar::xmom].nGrowVect());
183  const Box gtby = mfi.nodaltilebox(1).grow(S_old[IntVar::ymom].nGrowVect());
184  const Box gtbz = mfi.nodaltilebox(2).grow(S_old[IntVar::zmom].nGrowVect());
185 
186  Vector<Array4<Real> > sold_h(n_data);
187  Vector<Array4<Real> > snew_h(n_data);
188 
189  for (int i = 0; i < n_data; ++i) {
190  sold_h[i] = S_old[i].array(mfi);
191  snew_h[i] = S_new[i].array(mfi);
192  }
193 
194  Gpu::AsyncVector<Array4<Real> > sold_d(n_data);
195  Gpu::AsyncVector<Array4<Real> > snew_d(n_data);
196 
197  Gpu::copy(Gpu::hostToDevice, sold_h.begin(), sold_h.end(), sold_d.begin());
198  Gpu::copy(Gpu::hostToDevice, snew_h.begin(), snew_h.end(), snew_d.begin());
199 
200  Array4<Real>* sold = sold_d.dataPtr();
201  Array4<Real>* snew = snew_d.dataPtr();
202 
203  ParallelFor(gbx, static_cast<int>(Cons::NumVars),
204  [=] AMREX_GPU_DEVICE (int i, int j, int k, int n) {
205  snew[IntVar::cons](i,j,k,n) = sold[IntVar::cons](i,j,k,n);
206  });
207 
208  ParallelFor(gtbx, gtby, gtbz,
209  [=] AMREX_GPU_DEVICE (int i, int j, int k) noexcept {
210  snew[IntVar::xmom](i,j,k) = sold[IntVar::xmom](i,j,k);
211  },
212  [=] AMREX_GPU_DEVICE (int i, int j, int k) noexcept {
213  snew[IntVar::ymom](i,j,k) = sold[IntVar::ymom](i,j,k);
214  },
215  [=] AMREX_GPU_DEVICE (int i, int j, int k) noexcept {
216  snew[IntVar::zmom](i,j,k) = sold[IntVar::zmom](i,j,k);
217  });
218  }
219  }
220 
221  // Timestep taken by the fast integrator
222  amrex::Real dtau;
223 
224  // How many timesteps taken by the fast integrator
225  int nsubsteps;
226 
227  // We copy (rho theta) and the velocities here -- the velocities
228  // will be over-written in slow_rhs on all valid faces but we
229  // use this copy to fill in the ghost locations which will
230  // be needed for metric terms
231  #ifdef _OPENMP
232  #pragma omp parallel if (amrex::Gpu::notInLaunchRegion())
233  #endif
234  {
235  for ( MFIter mfi(S_old[IntVar::cons],TilingIfNotGPU()); mfi.isValid(); ++mfi) {
236  const Box gtbx = mfi.nodaltilebox(0).grow(S_old[IntVar::xmom].nGrowVect());
237  const Box gtby = mfi.nodaltilebox(1).grow(S_old[IntVar::ymom].nGrowVect());
238  const Box gtbz = mfi.nodaltilebox(2).grow(S_old[IntVar::zmom].nGrowVect());
239 
240  const Array4<Real>& scrh_xmom = (*S_scratch)[IntVar::xmom].array(mfi);
241  const Array4<Real>& scrh_ymom = (*S_scratch)[IntVar::ymom].array(mfi);
242  const Array4<Real>& scrh_zmom = (*S_scratch)[IntVar::zmom].array(mfi);
243 
244  const Array4<Real>& sold_xmom = S_old[IntVar::xmom].array(mfi);
245  const Array4<Real>& sold_ymom = S_old[IntVar::ymom].array(mfi);
246  const Array4<Real>& sold_zmom = S_old[IntVar::zmom].array(mfi);
247 
248  ParallelFor(gtbx, gtby, gtbz,
249  [=] AMREX_GPU_DEVICE (int i, int j, int k) noexcept {
250  scrh_xmom(i,j,k) = sold_xmom(i,j,k);
251  },
252  [=] AMREX_GPU_DEVICE (int i, int j, int k) noexcept {
253  scrh_ymom(i,j,k) = sold_ymom(i,j,k);
254  },
255  [=] AMREX_GPU_DEVICE (int i, int j, int k) noexcept {
256  scrh_zmom(i,j,k) = sold_zmom(i,j,k);
257  });
258  }
259  }
260 
261  for (int nrk = 0; nrk < 3; nrk++)
262  {
263  // amrex::Print() << "Starting RK3: Step " << nrk+1 << std::endl;
264  amrex::Real time_stage;
265  if (nrk == 0) { nsubsteps = 1; dtau = timestep / 3.0; time_stage = time + timestep / 3.0;}
266  if (nrk == 1) { nsubsteps = substep_ratio/2; dtau = sub_timestep ; time_stage = time + timestep / 2.0;}
267  if (nrk == 2) { nsubsteps = substep_ratio; dtau = sub_timestep ; time_stage = time + timestep ;}
268 
269  // step 1 starts with S_stage = S^n and we always start substepping at the old time
270  // step 2 starts with S_stage = S^* and we always start substepping at the old time
271  // step 3 starts with S_stage = S^** and we always start substepping at the old time
272 
273  pre_update(S_new, S_new[IntVar::cons].nGrow());
274 
275  // S_scratch also holds the average momenta over the fast iterations --
276  // to be used to update the slow variables -- we will initialize with
277  // the momenta used in the first call to the slow_rhs, then update
278  // inside fast_rhs, then use these values in the later call to slow_rhs
279 
280  // ****************************************************
281  // if (version == 0) evaluate F_slow(S_stage) only for the fast variables
282  // if (version == 1) evaluate F_slow(S_stage) only for all the variables
283  // ****************************************************
284  slow_rhs_pre(*F_slow, S_new, *S_scratch, time, time_stage);
285 
286  amrex::Real inv_fac = 1.0 / static_cast<amrex::Real>(nsubsteps);
287 
288  // ****************************************************
289  // Acoustic substepping
290  // ****************************************************
291  if (version == 0)
292  {
293  // *******************************************************************************
294  // Update the fast variables
295  // *******************************************************************************
296  for (int ks = 0; ks < nsubsteps; ++ks)
297  {
298  fast_rhs(ks, *F_slow, S_old, S_new, *S_sum, *S_scratch, dtau, inv_fac, time + (ks+1) * dtau);
299 
300  } // ks
301 
302  } else {
303  no_substep(*S_sum, S_old, *F_slow, time + nsubsteps*dtau, nsubsteps*dtau);
304  }
305 
306  // ****************************************************
307  // Evaluate F_slow(S_stage) only for the slow variables
308  // Note that we are using the current stage versions (in S_new) of the slow variables
309  // (because we didn't update the slow variables in the substepping)
310  // but we are using the "new" versions (in S_sum) of the velocities
311  // (because we did update the fast variables in the substepping)
312  // ****************************************************
313  slow_rhs_post(*F_slow, S_old, S_new, *S_sum, *S_scratch, time, time_stage);
314 
315  // Call the post-update hook for S_new after all the fast steps completed
316  // This will update S_prim that is used in the slow RHS
317  post_update(S_new, time + nsubsteps*dtau, dtau, S_new[IntVar::cons].nGrow(), S_new[IntVar::xmom].nGrow());
318  } // nrk
319 
320  // Return timestep
321  return timestep;
322  }
