76  {

80 
81  AMREX_ALWAYS_ASSERT(substep_ratio > 1 && substep_ratio % 2 == 0);
82 
83  const Real sub_timestep = BaseT::timestep / substep_ratio;
84 
85  // Assume before advance() that S_old is valid data at the current time ("time" argument)
86  // And that if data is a MultiFab, both S_old and S_new contain ghost cells for evaluating a stencil based RHS
87  // We need this from S_old. This is convenient for S_new to have so we can use it
88  // as scratch space for stage values without creating a new scratch MultiFab with ghost cells.
89 
90  // NOTE: In the following, we use S_new to hold S*, S**, and finally, S^(n+1) at the new time
91  // DEFINITIONS:
92  // S_old = S^n
93  // S_sum = S(t)
94  // F_slow = F(S_stage)
95  // F_pert = G(S(t)-S_stage, S_stage)
96 
97  /**********************************************/
98  /* RK3 Integration with Acoustuc Sub-stepping */
99  /**********************************************/
100 
101  // Start with S_new (aka S_stage) holding S_old
102  IntegratorOps<T>::Copy(S_new, S_old);
103  BaseT::post_update(S_new, time);
104 
105  // Timestep taken by the fast integrator
106  amrex::Real dtau;
107 
108  // How many timesteps taken by the fast integrator
109  int nsubsteps;
110 
111  for (int nrk = 0; nrk < 3; nrk++)
112  {
113  // amrex::Print() << "Starting RK3: Step " << nrk+1 << std::endl;
114  if (nrk == 0) { nsubsteps = 1; dtau = BaseT::timestep / 3.0_rt; }
115  if (nrk == 1) { nsubsteps = substep_ratio/2; dtau = sub_timestep;}
116  if (nrk == 2) { nsubsteps = substep_ratio; dtau = sub_timestep;}
117 
118  // step 1 starts with S_stage = S^n and we always start substepping at the old time
119  // step 2 starts with S_stage = S^* and we always start substepping at the old time
120  // step 3 starts with S_stage = S^** and we always start substepping at the old time
121  IntegratorOps<T>::Copy(*S_sum , S_old);
122  IntegratorOps<T>::Copy(*S_sum_old, S_old);
123 
124  // Evaluate F_slow(S_stage)
125  BaseT::rhs(*F_slow, S_new, time);
126 
127  for (int k = 0; k < nsubsteps; ++k)
128  {
129  // Evaluate F_pert(S_pert, S_old)
130  // S_sum is used in the fast RHS to internally define S_pert = (S_sum - S_stage)
131  implicit_fast_rhs(*F_pert, *F_slow, S_new, *S_sum, *S_sum_old, time + k*dtau, dtau);
132 
133  // Update S_sum = S_pert + S_stage
134  IntegratorOps<T>::Saxpy(*S_sum, dtau, *F_slow);
135  IntegratorOps<T>::Saxpy(*S_sum, dtau, *F_pert);
136 
137  IntegratorOps<T>::Copy(*S_sum_old, *S_sum);
138 
139  // Call the post-substep hook for S_sum at t = time + (k+1) * sub_dt
140  post_substep(*S_sum , time + (k+1) * dtau);
141  post_substep(*S_sum_old, time + (k+1) * dtau);
142  }
143 
144  // Define the solution at the next stage
145  IntegratorOps<T>::Copy(S_new, *S_sum);
146 
147  // Call the post-update hook for S_new after all the fine steps completed
148  BaseT::post_update(S_new, time + nsubsteps*dtau);
149  }
150 
151  // Return timestep
152  return BaseT::timestep;
153  }
