130  amrex::Real advance (T& S_old, T& S_new, amrex::Real time, const amrex::Real time_step)
131  {
132  timestep = time_step;
133 
134  const int substep_ratio = get_slow_fast_timestep_ratio();
135 
136  AMREX_ALWAYS_ASSERT(substep_ratio > 1 && substep_ratio % 2 == 0);
137 
138  const amrex::Real sub_timestep = timestep / substep_ratio;
139 
140  // Assume before advance() that S_old is valid data at the current time ("time" argument)
141  // And that if data is a MultiFab, both S_old and S_new contain ghost cells for evaluating a stencil based RHS
142  // We need this from S_old. This is convenient for S_new to have so we can use it
143  // as scratch space for stage values without creating a new scratch MultiFab with ghost cells.
144 
145  // NOTE: In the following, we use S_new to hold S*, S**, and finally, S^(n+1) at the new time
146  // DEFINITIONS:
147  // S_old = S^n
148  // S_sum = S(t)
149  // F_slow = F(S_stage)
150  // F_pert = G(S(t)-S_stage, S_stage)
151 
152  /**********************************************/
153  /* RK3 Integration with Acoustic Sub-stepping */
154  /**********************************************/
155 
156  // Start with S_new (aka S_stage) holding S_old
157  amrex::IntegratorOps<T>::Copy(S_new, S_old);
158  post_update(S_new, time);
159 
160  // Timestep taken by the fast integrator
161  amrex::Real dtau;
162 
163  // How many timesteps taken by the fast integrator
164  int nsubsteps;
165 
166  int nav = Cons::NumVars;
167  const amrex::Vector<int> scomp_all = {0,0,0,0,0,0,0};
168  const amrex::Vector<int> ncomp_all = {nav,1,1,1,nav,nav,nav};
169 
170  const amrex::Vector<int> scomp_fast = {0,0,0,0,0,0,0};
171  const amrex::Vector<int> ncomp_fast = {2,1,1,1,2,2,2};
172 
173  int nsv = Cons::NumVars-2;
174  const amrex::Vector<int> scomp_slow = { 2,0,0,0, 2, 2, 2};
175  const amrex::Vector<int> ncomp_slow = {nsv,0,0,0,nsv,nsv,nsv};
176 
177  // We copy (rho theta) and the velocities here -- the velocities
178  // will be over-written in slow_rhs on all valid faces but we
179  // use this copy to fill in the ghost locations which will
180  // be needed for metric terms
181  const amrex::Vector<int> scomp_vels = {0,0,0,0,0,0,0};
182  const amrex::Vector<int> ncomp_vels = {0,1,1,1,0,0,0};
183  amrex::IntegratorOps<T>::Copy(*S_scratch, S_old, scomp_vels, ncomp_vels);
184 
185 
186  for (int nrk = 0; nrk < 3; nrk++)
187  {
188  // amrex::Print() << "Starting RK3: Step " << nrk+1 << std::endl;
189  if (nrk == 0) { nsubsteps = 1; dtau = timestep / 3.0; }
190  if (nrk == 1) { nsubsteps = substep_ratio/2; dtau = sub_timestep;}
191  if (nrk == 2) { nsubsteps = substep_ratio; dtau = sub_timestep;}
192 
193  // step 1 starts with S_stage = S^n and we always start substepping at the old time
194  // step 2 starts with S_stage = S^* and we always start substepping at the old time
195  // step 3 starts with S_stage = S^** and we always start substepping at the old time
196 
197  // We fill the fast variables with the data from the start of the RK iteration, while
198  // we fill the slow variables with the most recent RK stage data
199  amrex::IntegratorOps<T>::Copy(*S_sum , S_old, scomp_fast, ncomp_fast);
200  amrex::IntegratorOps<T>::Copy(*S_sum , S_new, scomp_slow, ncomp_slow);
201 
202  // S_scratch will hold (rho theta) from the previous fast timestep
203  const amrex::Vector<int> scomp_rth = {1,0,0,0,0,0,0};
204  const amrex::Vector<int> ncomp_rth = {1,0,0,0,0,0,0};
205  amrex::IntegratorOps<T>::Copy(*S_scratch, S_old, scomp_rth, ncomp_rth);
206 
207  // S_scratch also holds the average momenta over the fast iterations --
208  // to be used to update the slow variables -- we will initialize with
209  // the momenta used in the first call to the slow_rhs, then update
210  // inside fast_rhs, then use these values in the later call to slow_rhs
211 
212  // Evaluate F_slow(S_stage) only for the fast variables
213  slow_rhs(*F_slow, S_new, *S_scratch, time, RHSVar::fast);
214 
215  amrex::Real inv_fac = 1.0 / static_cast<amrex::Real>(nsubsteps);
216 
217  for (int k = 0; k < nsubsteps; ++k)
218  {
219  // Evaluate F_fast(S_pert, S_old) on the fast variables
220  // S_sum is used in the fast RHS to internally define S_pert = (S_sum - S_stage)
221  fast_rhs(*F_pert, *F_slow, S_new, *S_sum, *S_scratch, dtau, inv_fac);
222 
223  // Update S_sum = S_pert + S_stage only for the fast variables
224  amrex::IntegratorOps<T>::Saxpy(*S_sum, dtau, *F_slow, scomp_fast, ncomp_fast);
225  amrex::IntegratorOps<T>::Saxpy(*S_sum, dtau, *F_pert, scomp_fast, ncomp_fast);
226 
227  // Call the post-substep hook for S_sum at t = time + (k+1) * sub_dt
228  // Only call the post_substep on the fast variables
229  post_substep(*S_sum , time + (k+1) * dtau);
230 
231  // Copy only (rho theta), including ghost cells
232  amrex::IntegratorOps<T>::Copy(*S_scratch, *S_sum, scomp_rth, ncomp_rth, true);
233  }
234 
235  // Evaluate F_slow(S_stage) only for the slow variables
236  // Note that we are using the current stage versions (in S_new) of the slow variables
237  // (because we didn't update the slow variables in the substepping)
238  // but we are using the "new" versions (in S_sum) of the velocities
239  // (because we did update the fast variables in the substepping)
240  slow_rhs(*F_slow, *S_sum, *S_scratch, time, RHSVar::slow);
241 
242  // Apply F_slow only for the slow variables
243  amrex::IntegratorOps<T>::Copy(*S_sum , S_old, scomp_slow, ncomp_slow);
244  amrex::IntegratorOps<T>::Saxpy(*S_sum, nsubsteps*dtau, *F_slow, scomp_slow, ncomp_slow);
245 
246  // Copy into S_new which holds the stage data
247  amrex::IntegratorOps<T>::Copy(S_new, *S_sum, scomp_all, ncomp_all);
248 
249  // Call the post-update hook for S_new after all the fine steps completed
250  // This will update S_prim that is used in the slow RHS
251  post_update(S_new, time + nsubsteps*dtau);
252  }
253 
254  // Return timestep
255  return timestep;
256  }
