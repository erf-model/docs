452  coriolis_factor = 2.0 * 2.0 * PI / rot_time_period;
453  amrex::Print() << "Coriolis factor = " << coriolis_factor << std::endl;
454 
455  amrex::Real latitude = 90.0;
456  pp.query("latitude", latitude);
457 
458  pp.query("coriolis_3d", coriolis_3d);
459 
460  // Convert to radians
461  latitude *= (PI/180.);
462  sinphi = std::sin(latitude);
463  if (coriolis_3d) {
464  cosphi = std::cos(latitude);
465  }
466 
467  amrex::Print() << "Coriolis frequency, f = " << coriolis_factor * sinphi << " 1/s" << std::endl;
468 
469  if (abl_driver_type == ABLDriverType::GeostrophicWind) {
470  // Read in the geostrophic wind -- we only use this to construct
471  // the forcing term so no need to keep it
472  amrex::Vector<amrex::Real> abl_geo_wind(3);
473  pp.queryarr("abl_geo_wind",abl_geo_wind);
474 
475  if(!pp.query("abl_geo_wind_table",abl_geo_wind_table)) {
476  abl_geo_forcing = {
477  -coriolis_factor * (abl_geo_wind[1]*sinphi - abl_geo_wind[2]*cosphi),
478  coriolis_factor * abl_geo_wind[0]*sinphi,
479  -coriolis_factor * abl_geo_wind[0]*cosphi
