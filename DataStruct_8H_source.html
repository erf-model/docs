341  void build_coriolis_forcings()
342  {
343  amrex::ParmParse pp(pp_prefix);
344 
345  // Read the rotational time period (in seconds)
346  amrex::Real rot_time_period = 86400.0;
347  pp.query("rotational_time_period", rot_time_period);
348 
349  coriolis_factor = 2.0 * 2.0 * PI / rot_time_period;
350  amrex::Print() << "Coriolis factor = " << coriolis_factor << std::endl;
351 
352  amrex::Real latitude = 90.0;
353  pp.query("latitude", latitude);
354 
355  // Convert to radians
356  latitude *= (PI/180.);
357  sinphi = std::sin(latitude);
358  cosphi = std::cos(latitude);
359 
360  if (abl_driver_type == ABLDriverType::GeostrophicWind)
361  {
362  // Read in the geostrophic wind -- we only use this to construct
363  // the forcing term so no need to keep it
364  amrex::Vector<amrex::Real> abl_geo_wind(3);
365  pp.queryarr("abl_geo_wind",abl_geo_wind);
366 
367  abl_geo_forcing = {
368  -coriolis_factor * (abl_geo_wind[1]*sinphi - abl_geo_wind[2]*cosphi),
369  coriolis_factor * abl_geo_wind[0]*sinphi,
370  -coriolis_factor * abl_geo_wind[0]*cosphi
371  };
372  }
