473  void read_from_file(const std::string input_sounding_file,
474  const amrex::Real ztop)
475  {
476  z_inp_sound.resize(0);
477  theta_inp_sound.resize(0);
478  qv_inp_sound.resize(0);
479  U_inp_sound.resize(0);
480  V_inp_sound.resize(0);
481 
482  pm_integ.resize(0);
483  rhom_integ.resize(0);
484 
485  pd_integ.resize(0);
486  rhod_integ.resize(0);
487 
488  // Read the input_sounding file
489  amrex::Print() << "input_sounding file location : " << input_sounding_file << std::endl;
490  std::ifstream input_sounding_reader(input_sounding_file);
491  if(!input_sounding_reader.is_open()) {
492  amrex::Error("Error opening the input_sounding file\n");
493  }
494  else {
495  // Read the contents of the input_sounding file
496  amrex::Print() << "Successfully opened the input_sounding file. Now reading... " << std::endl;
497  std::string line;
498 
499  // Read the first line
500  std::getline(input_sounding_reader, line);
501  std::istringstream iss(line);
502  iss >> press_ref_inp_sound >> theta_ref_inp_sound >> qv_ref_inp_sound;
503  press_ref_inp_sound *= 100; // convert from hPa to Pa
504  qv_ref_inp_sound *= 0.001; // convert from g/kg to kg/kg
505 
506  // Add surface
507  z_inp_sound.push_back(0);
508  theta_inp_sound.push_back(theta_ref_inp_sound);
509  qv_inp_sound.push_back(qv_ref_inp_sound);
510  U_inp_sound.push_back(0);
511  V_inp_sound.push_back(0);
512 
513  // Read the vertical profile at each given height
514  amrex::Real z, theta, qv, U, V;
515  while(std::getline(input_sounding_reader, line)) {
516  std::istringstream iss_z(line);
517  iss_z >> z >> theta >> qv >> U >> V;
518  if (z == 0) {
519  AMREX_ALWAYS_ASSERT(theta == theta_inp_sound[0]);
520  AMREX_ALWAYS_ASSERT(qv == qv_inp_sound[0]);
521  U_inp_sound[0] = U;
522  V_inp_sound[0] = V;
523  } else {
524  AMREX_ALWAYS_ASSERT(z > z_inp_sound[0]);
525  z_inp_sound.push_back(z);
526  theta_inp_sound.push_back(theta);
527  qv_inp_sound.push_back(qv*0.001);
528  U_inp_sound.push_back(U);
529  V_inp_sound.push_back(V);
530  if (z >= ztop) break;
531  }
532  }
533 
534  // At this point, we have an input_sounding with
535  // z_inp_sound[N-1] <= ztop
536  if (z != ztop)
537  {
538  int N = size();
539  amrex::Real z1 = z_inp_sound[N-2];
540  amrex::Real th1 = theta_inp_sound[N-2];
541  amrex::Real qv1 = qv_inp_sound[N-2];
542  amrex::Real U1 = U_inp_sound[N-2];
543  amrex::Real V1 = V_inp_sound[N-2];
544  amrex::Real z2 = z_inp_sound[N-1];
545  amrex::Real th2 = theta_inp_sound[N-1];
546  amrex::Real qv2 = qv_inp_sound[N-1];
547  amrex::Real U2 = U_inp_sound[N-1];
548  amrex::Real V2 = V_inp_sound[N-1];
549 
550  amrex::Real zfrac = (ztop - z1) / (z2 - z1);
551  amrex::Real th_top = th1 + zfrac*(th2 - th1);
552  amrex::Real qv_top = qv1 + zfrac*(qv2 - qv1);
553  amrex::Real U_top = U1 + zfrac*( U2 - U1);
554  amrex::Real V_top = V1 + zfrac*( V2 - V1);
555 
556  if (z < ztop)
557  {
558  amrex::Print() << "Note: sounding zmax=" << z
559  << ", extrapolating to zhi=" << ztop << std::endl;
560  // Add domain top with extrapolated values
561  z_inp_sound.push_back(ztop);
562  theta_inp_sound.push_back(th_top);
563  qv_inp_sound.push_back(qv_top);
564  U_inp_sound.push_back(U_top);
565  V_inp_sound.push_back(V_top);
566  }
567  else // z > ztop
568  {

586  void calc_rho_p(const amrex::Real ztop)
587  {
588  /* Calculate density and pressure, roughly following the procedure in
589  * WRF dyn_em/module_initialize_ideal.F. We integrate hydrostatically
590  * from the surface up through the air column to get the total (moist)
591  * density and pressure, then back down to get the dry density and
592  * pressure.
593  *
594  * This deviates from that code slightly because we calculate the
595  * correct moist theta, equal to virtual potential temperature, and
596  * also remove the factor (1+qv) for consistency with the the surface
597  * density.
598  */
599  const int maxiter = 10;
600  const int Ninp = size();
601  pm_integ.resize(Ninp);
602  pd_integ.resize(Ninp);
603  rhom_integ.resize(Ninp);
604  rhod_integ.resize(Ninp);
605 
606  // evaluate surface quantities (k=0)
607  amrex::Real thm_surf = theta_ref_inp_sound *
608  (1 + (R_v/R_d-1) * qv_ref_inp_sound); // _moist_ theta == virtual potential temperature
609  pm_integ[0] = press_ref_inp_sound; // _total_ pressure (incl moisture)
610  rhom_integ[0] = 1 / (
611  R_d/p_0 * thm_surf * std::pow(pm_integ[0]/p_0, -iGamma)); // density of _moist_ air
612 
613  amrex::Print() << "ideal sounding init: surface density of moist air = "
614  << rhom_integ[0] << " kg/m^3" << std::endl;
615  amrex::Print() << "integrating moist air column up to " << ztop << std::endl;
616 
617  // Note:
618  // p_dry = rho_d R_d T
619  // p_tot = rho_m R_d T_v
620  // = rho_d(1 + q_v) R_d T_v
621 
622  // integrate from surface to domain top
623  amrex::Real qvf, dz;
624  for (int k=1; k < size(); ++k)
625  {
626  qvf = 1 + (R_v/R_d-1) * qv_inp_sound[k];
627  dz = z_inp_sound[k] - z_inp_sound[k-1];
628  rhom_integ[k] = rhom_integ[k-1]; // guess
629  for (int it=0; it < maxiter; ++it)
630  {
631  pm_integ[k] = pm_integ[k-1]
632  - 0.5*dz*(rhom_integ[k] + rhom_integ[k-1])*CONST_GRAV;
633  AMREX_ALWAYS_ASSERT(pm_integ[k] > 0);
634  rhom_integ[k] = 1 / (
635  R_d/p_0 * theta_inp_sound[k]*qvf * std::pow(pm_integ[k]/p_0, -iGamma));
636  }
637  }
638 
639  // now, integrate from the top of the sounding (where it's dry) back
640  // down to get the dry air column properties
641  pd_integ[Ninp-1] = pm_integ[Ninp-1];
642  rhod_integ[Ninp-1] = 1 / (
643  R_d/p_0 * theta_inp_sound[Ninp-1] * std::pow(pd_integ[Ninp-1]/p_0, -iGamma));
644  amrex::Print() << "z p_d rho_d theta" << std::endl;
645  amrex::Print() << z_inp_sound[Ninp-1] << " " << pd_integ[Ninp-1] << " " << rhod_integ[Ninp-1] << " " << theta_inp_sound[Ninp-1] << std::endl;
646  for (int k=Ninp-2; k >= 0; --k)
647  {
648  dz = z_inp_sound[k+1] - z_inp_sound[k];
649  rhod_integ[k] = rhod_integ[k+1]; // guess
650  for (int it=0; it < maxiter; ++it)
651  {
652  pd_integ[k] = pd_integ[k+1]
653  + 0.5*dz*(rhod_integ[k] + rhod_integ[k+1])*CONST_GRAV;
654  AMREX_ALWAYS_ASSERT(pd_integ[k] > 0);
655  rhod_integ[k] = 1 / (
656  R_d/p_0 * theta_inp_sound[k] * std::pow(pd_integ[k]/p_0, -iGamma));
657  }
658  amrex::Print() << z_inp_sound[k] << " " << pd_integ[k] << " " << rhod_integ[k] << " " << theta_inp_sound[k] << std::endl; // DEBUG
659  }
660  // Note: at this point, the surface pressure, density of the dry air
661  // column is stored in pd_integ[0], rhod_integ[0]
662 
663  // update
664  host_to_device();
665  }
