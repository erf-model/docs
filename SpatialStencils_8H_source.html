65 // This routine follows WFromOmega with clipping at the boundary:
66 // Terrain metrics at the boundary are used for ghost cells (0 gradient).

69 amrex::Real WFromOmegaBC(int i, int j, int k, amrex::Real omega,
70  const amrex::Array4<const amrex::Real>& u,
71  const amrex::Array4<const amrex::Real>& v,
72  const amrex::Array4<const amrex::Real>& z_nd,
73  const amrex::GpuArray<amrex::Real, AMREX_SPACEDIM>& dxInv)
74 {
75  // Clip metric indices for ghost cell BCs
76  amrex::Dim3 hi = ubound(z_nd);
77  amrex::Dim3 lo = lbound(z_nd);
78  int im = (i >= hi.x) ? (hi.x-1) : i;
79  im = (im <= lo.x) ? (lo.x+1) : im;
80  int jm = (j >= hi.y) ? (hi.y-1) : j;
81  jm = (jm <= lo.y) ? (lo.y+1) : jm;
82  int km = (k >= hi.z) ? (hi.z-1) : k;
83  km = (km <= lo.z) ? (lo.z+1) : km;
84 
85  // This is dh/dxi at z-face (i,j,k-1/2)
86  amrex::Real met_zlo_xi = 0.5 * dxInv[0] *
87  ( z_nd(im+1,jm+1,km ) + z_nd(im+1,jm ,km ) // hi i, lo k
88  -z_nd(im ,jm+1,km ) - z_nd(im ,jm ,km ) ); // lo i, lo k
89 
90  // This is dh/deta at z-face (i,j,k-1/2)
91  amrex::Real met_zlo_eta = 0.5 * dxInv[1] *
92  ( z_nd(im+1,jm+1,km ) + z_nd(im ,jm+1,km ) // hi j, lo k
93  -z_nd(im+1,jm ,km ) - z_nd(im ,jm ,km ) ); // lo j, lo k
94 
95  // Clip velocity indices for ghost cell BCs
96  amrex::Real vec_zlo_xi;
97  amrex::Real vec_zlo_eta;
98  hi = ubound(u);
99  lo = lbound(u);
100  hi.z = hi.z + 1; // nodal for zvelocity
101 
102  if( k == hi.z) {
103  // foextrap to highest k-face
104  vec_zlo_xi = 0.75*( u(i,j,k-1)+u(i+1,j,k-1) ) - 0.25*( u(i,j,k-2)+u(i+1,j,k-2) );
105  vec_zlo_eta = 0.75*( v(i,j,k-1)+v(i,j+1,k-1) ) - 0.25*( v(i,j,k-2)+v(i,j+1,k-2) );
106  } else if (k == lo.z) {
107  // foextrap to lowest k-face
108  vec_zlo_xi = 0.75*( u(i,j,k)+u(i+1,j,k) ) - 0.25*( u(i,j,k+1)+u(i+1,j,k+1) );
109  vec_zlo_eta = 0.75*( v(i,j,k)+v(i,j+1,k) ) - 0.25*( v(i,j,k+1)+v(i,j+1,k+1) );
110  } else {
111  // Valid ghost cell for u & v
112  vec_zlo_xi = 0.25 * ( u(i,j,k-1) + u(i+1,j,k-1) + u(i,j,k) + u(i+1,j,k) );
113  vec_zlo_eta = 0.25 * ( v(i,j,k-1) + v(i,j+1,k-1) + v(i,j,k) + v(i,j+1,k) );
114  }
115 
116  amrex::Real w = met_zlo_xi * vec_zlo_xi + met_zlo_eta * vec_zlo_eta + omega;
117  return w;
118 }
