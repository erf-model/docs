456  Array4<Real> diffflux_y = dflux_y->array(mfi);

458

465 
466  // NOTE: No diffusion for continuity, so n starts at 1.
467  int n_start = amrex::max(start_comp,RhoTheta_comp);
468  int n_comp = end_comp - n_start + 1;
469

472  cell_data, cell_prim, cell_rhs,
473  diffflux_x, diffflux_y, diffflux_z,
474  z_nd, ax_arr, ay_arr, az_arr, detJ_arr,
475  dxInv, SmnSmn_a, mf_m, mf_u, mf_v,
476  hfx_z, q1fx_z, q2fx_z, diss, mu_turb, dc, tc,
477  tm_arr, grav_gpu, bc_ptr_d, l_use_most);
478  } else {
479  DiffusionSrcForState_N(bx, domain, n_start, n_comp, l_exp_most, u, v,
480  cell_data, cell_prim, cell_rhs,
481  diffflux_x, diffflux_y, diffflux_z,
482  dxInv, SmnSmn_a, mf_m, mf_u, mf_v,
483  hfx_z, q1fx_z, q2fx_z, diss,
484  mu_turb, dc, tc,
485  tm_arr, grav_gpu, bc_ptr_d, l_use_most);
486  }
487  }
488 
489  const Array4<Real const>& source_arr = cc_src.const_array(mfi);
490  ParallelFor(bx, [=] AMREX_GPU_DEVICE (int i, int j, int k) noexcept
491  {
492  cell_rhs(i,j,k,Rho_comp) += source_arr(i,j,k,Rho_comp);
493  cell_rhs(i,j,k,RhoTheta_comp) += source_arr(i,j,k,RhoTheta_comp);
494  });
495 
496  // Multiply the slow RHS for rho and rhotheta by detJ here so we don't have to later
497  if (l_use_terrain && l_moving_terrain) {
498  ParallelFor(bx, [=] AMREX_GPU_DEVICE (int i, int j, int k) noexcept
499  {
500  cell_rhs(i,j,k,Rho_comp) *= detJ_arr(i,j,k);
501  cell_rhs(i,j,k,RhoTheta_comp) *= detJ_arr(i,j,k);
502  });
503  }
504 
505  // If incompressible and in second RK stage, take average of old-time and new-time source
506  if ( l_incompressible && (nrk == 1) )
507  {
508  ParallelFor(bx, [=] AMREX_GPU_DEVICE (int i, int j, int k) noexcept
509  {
510  cell_rhs(i,j,k, Rho_comp) *= 0.5;
511  cell_rhs(i,j,k,RhoTheta_comp) *= 0.5;
512 
513  cell_rhs(i,j,k, Rho_comp) += 0.5 / dt * (cell_data(i,j,k, Rho_comp) - cell_old(i,j,k, Rho_comp));
514  cell_rhs(i,j,k,RhoTheta_comp) += 0.5 / dt * (cell_data(i,j,k,RhoTheta_comp) - cell_old(i,j,k,RhoTheta_comp));
515  });
516  }
517 
518  // *****************************************************************************
519  // Define updates in the RHS of {x, y, z}-momentum equations
520  // *****************************************************************************
521  int lo_z_face;
522  int hi_z_face;
523  if (level == 0) {
524  lo_z_face = domain.smallEnd(2);
525  hi_z_face = domain.bigEnd(2)+1;
526  } else {
527  lo_z_face = mfi.validbox().smallEnd(2);
528  hi_z_face = mfi.validbox().bigEnd(2)+1;
529  }
530  AdvectionSrcForMom(bx, tbx, tby, tbz,
531  rho_u_rhs, rho_v_rhs, rho_w_rhs,
532  cell_data, u, v, w,
533  rho_u, rho_v, omega_arr,
534  z_nd, ax_arr, ay_arr, az_arr, detJ_arr,
535  dxInv, mf_m, mf_u, mf_v,
536  l_horiz_adv_type, l_vert_adv_type,
537  l_horiz_upw_frac, l_vert_upw_frac,
538  l_use_terrain, lo_z_face, hi_z_face,
539  domain, bc_ptr_h);
540 
541  if (l_use_diff) {
542  // Note: tau** were calculated with calls to
543  // ComputeStress[Cons|Var]Visc_[N|T] in which ConsVisc ("constant
544  // viscosity") means that there is no contribution from a
545  // turbulence model. However, whether this field truly is constant
546  // depends on whether MolecDiffType is Constant or ConstantAlpha.
547  if (l_use_terrain) {
548  DiffusionSrcForMom_T(tbx, tby, tbz,
549  rho_u_rhs, rho_v_rhs, rho_w_rhs,
550  tau11, tau22, tau33,
551  tau12, tau13,
552  tau21, tau23,
553  tau31, tau32,
554  detJ_arr, dxInv,
555  mf_m, mf_u, mf_v);
556  } else {
557  DiffusionSrcForMom_N(tbx, tby, tbz,
558  rho_u_rhs, rho_v_rhs, rho_w_rhs,
559  tau11, tau22, tau33,
560  tau12, tau13, tau23,
561  dxInv,
562  mf_m, mf_u, mf_v);
563  }
564  }
565 
566  auto abl_pressure_grad = solverChoice.abl_pressure_grad;
567 
568  ParallelFor(tbx, [=] AMREX_GPU_DEVICE (int i, int j, int k)
569  { // x-momentum equation
570 
571  //Note : mx/my == 1, so no map factor needed here
572  Real gp_xi = dxInv[0] * (pp_arr(i,j,k) - pp_arr(i-1,j,k));
573  Real gpx = gp_xi;
574 
575  if (l_use_terrain) {
576  Real met_h_xi = Compute_h_xi_AtIface (i, j, k, dxInv, z_nd);
577  Real met_h_zeta = Compute_h_zeta_AtIface(i, j, k, dxInv, z_nd);
578 
579  Real gp_zeta_on_iface;
580  if (k==0) {
581  gp_zeta_on_iface = 0.5 * dxInv[2] * (
582  pp_arr(i-1,j,k+1) + pp_arr(i,j,k+1)
583  - pp_arr(i-1,j,k ) - pp_arr(i,j,k ) );
584  } else if (k==domhi_z) {
585  gp_zeta_on_iface = 0.5 * dxInv[2] * (
586  pp_arr(i-1,j,k ) + pp_arr(i,j,k )
587  - pp_arr(i-1,j,k-1) - pp_arr(i,j,k-1) );
588  } else {
589  gp_zeta_on_iface = 0.25 * dxInv[2] * (
590  pp_arr(i-1,j,k+1) + pp_arr(i,j,k+1)
591  - pp_arr(i-1,j,k-1) - pp_arr(i,j,k-1) );
592  }
593  gpx -= (met_h_xi/ met_h_zeta) * gp_zeta_on_iface;
594  }
595 
596  gpx *= mf_u(i,j,0);
597 
598  Real q = 0.0;
599  if (l_use_moisture) {
600  q = 0.5 * ( cell_prim(i,j,k,PrimQ1_comp) + cell_prim(i-1,j,k,PrimQ1_comp)
601  +cell_prim(i,j,k,PrimQ2_comp) + cell_prim(i-1,j,k,PrimQ2_comp) );
602  }
603 
604  rho_u_rhs(i, j, k) += (-gpx - abl_pressure_grad[0]) / (1.0 + q)
605  + xmom_src_arr(i,j,k);
606 
607  if (l_moving_terrain) {
608  Real h_zeta = Compute_h_zeta_AtIface(i, j, k, dxInv, z_nd);
609  rho_u_rhs(i, j, k) *= h_zeta;
610  }
611 
612  if ( l_incompressible && (nrk == 1) ) {
613  rho_u_rhs(i,j,k) *= 0.5;
614  rho_u_rhs(i,j,k) += 0.5 / dt * (rho_u(i,j,k) - rho_u_old(i,j,k));
615  }
616  });
617 
618  ParallelFor(tby, [=] AMREX_GPU_DEVICE (int i, int j, int k)
619  { // y-momentum equation
620 
621  //Note : mx/my == 1, so no map factor needed here
622  Real gp_eta = dxInv[1] * (pp_arr(i,j,k) - pp_arr(i,j-1,k));
623  Real gpy = gp_eta;
624 
625  if (l_use_terrain) {
626  Real met_h_eta = Compute_h_eta_AtJface (i, j, k, dxInv, z_nd);
627  Real met_h_zeta = Compute_h_zeta_AtJface(i, j, k, dxInv, z_nd);
628  Real gp_zeta_on_jface;
629  if(k==0) {
630  gp_zeta_on_jface = 0.5 * dxInv[2] * (
631  pp_arr(i,j,k+1) + pp_arr(i,j-1,k+1)
632  - pp_arr(i,j,k ) - pp_arr(i,j-1,k ) );
633  } else if (k==domhi_z) {
634  gp_zeta_on_jface = 0.5 * dxInv[2] * (
635  pp_arr(i,j,k ) + pp_arr(i,j-1,k )
636  - pp_arr(i,j,k-1) - pp_arr(i,j-1,k-1) );
637  } else {
638  gp_zeta_on_jface = 0.25 * dxInv[2] * (
639  pp_arr(i,j,k+1) + pp_arr(i,j-1,k+1)
640  - pp_arr(i,j,k-1) - pp_arr(i,j-1,k-1) );
641  }
642  gpy -= (met_h_eta / met_h_zeta) * gp_zeta_on_jface;
643  } // l_use_terrain
644 
645  gpy *= mf_v(i,j,0);
646 
647  Real q = 0.0;
648  if (l_use_moisture) {
649  q = 0.5 * ( cell_prim(i,j,k,PrimQ1_comp) + cell_prim(i,j-1,k,PrimQ1_comp)
650  +cell_prim(i,j,k,PrimQ2_comp) + cell_prim(i,j-1,k,PrimQ2_comp) );
651  }
652 
653  rho_v_rhs(i, j, k) += (-gpy - abl_pressure_grad[1]) / (1.0_rt + q) + ymom_src_arr(i,j,k);
654 
655  if (l_moving_terrain) {
656  Real h_zeta = Compute_h_zeta_AtJface(i, j, k, dxInv, z_nd);
657  rho_v_rhs(i, j, k) *= h_zeta;
658  }
659 
660  if ( l_incompressible && (nrk == 1) ) {
661  rho_v_rhs(i,j,k) *= 0.5;
662  rho_v_rhs(i,j,k) += 0.5 / dt * (rho_v(i,j,k) - rho_v_old(i,j,k));
663  }
664  });
665 
666  // *****************************************************************************
667  // Zero out source terms for x- and y- momenta if at walls or inflow
668  // We need to do this -- even though we call the boundary conditions later --
669  // because the slow source is used to update the state in the fast interpolater.
670  // *****************************************************************************
671  {
672  if ( (bx.smallEnd(0) == domain.smallEnd(0)) &&
673  (bc_ptr_h[BCVars::xvel_bc].lo(0) == ERFBCType::ext_dir) ) {

676  rho_u_rhs(i,j,k) = 0.;
677  });
678  }
679  if ( (bx.bigEnd(0) == domain.bigEnd(0)) &&

681  Box hi_x_dom_face(bx); hi_x_dom_face.setSmall(0,bx.bigEnd(0)+1); hi_x_dom_face.setBig(0,bx.bigEnd(0)+1);
682  ParallelFor(hi_x_dom_face, [=] AMREX_GPU_DEVICE (int i, int j, int k) {
683  rho_u_rhs(i,j,k) = 0.;
684  });
685  }
686  if ( (bx.smallEnd(1) == domain.smallEnd(1)) &&
687  (bc_ptr_h[BCVars::yvel_bc].lo(1) == ERFBCType::ext_dir) ) {

689  ParallelFor(lo_y_dom_face, [=] AMREX_GPU_DEVICE (int i, int j, int k) {
690  rho_v_rhs(i,j,k) = 0.;
691  });
692  }
693  if ( (bx.bigEnd(1) == domain.bigEnd(1)) &&
694  (bc_ptr_h[BCVars::yvel_bc].hi(1) == ERFBCType::ext_dir) ) {

696  ParallelFor(hi_y_dom_face, [=] AMREX_GPU_DEVICE (int i, int j, int k) {
697  rho_v_rhs(i,j,k) = 0.;
698  });
699  }
700  }
701 
702  // *****************************************************************************
703  // Zero out source term for z-momentum at top and bottom of grid
704  // *****************************************************************************
705  Box b2d = tbz;
706  b2d.setSmall(2,lo_z_face);
707  b2d.setBig(2,lo_z_face);
708  // Enforce no forcing term at top and bottom boundaries of this grid
709  // We do this even when not at top or bottom of the domain because
710  // z-vel on the coarse/fine boundary is given by the coarse value
711  // (suitably interpolated tangentially and in time)
712  ParallelFor(b2d, [=] AMREX_GPU_DEVICE (int i, int j, int) {
713  rho_w_rhs(i,j,lo_z_face) = 0.;
714  rho_w_rhs(i,j,hi_z_face) = 0.;
715  });
716 
717  ParallelFor(tbz, [=] AMREX_GPU_DEVICE (int i, int j, int k)
718  { // z-momentum equation
719 
720  Real met_h_zeta = (l_use_terrain) ? Compute_h_zeta_AtKface(i, j, k, dxInv, z_nd) : 1;
721  Real gpz = dxInv[2] * ( pp_arr(i,j,k)-pp_arr(i,j,k-1) ) / met_h_zeta;
722 
723  Real q = 0.0;
724  if (l_use_moisture) {
725  q = 0.5 * ( cell_prim(i,j,k,PrimQ1_comp) + cell_prim(i,j,k-1,PrimQ1_comp)
726  +cell_prim(i,j,k,PrimQ2_comp) + cell_prim(i,j,k-1,PrimQ2_comp) );
727  }
728  rho_w_rhs(i, j, k) += (zmom_src_arr(i,j,k) - gpz - abl_pressure_grad[2]) / (1.0_rt + q);
729 
730  if (l_use_terrain && l_moving_terrain) {
731  rho_w_rhs(i, j, k) *= 0.5 * (detJ_arr(i,j,k) + detJ_arr(i,j,k-1));
732  }
733  });
734 
735  {
736  BL_PROFILE("slow_rhs_pre_fluxreg");
737  // We only add to the flux registers in the final RK step
738  // NOTE: for now we are only refluxing density not (rho theta) since the latter seems to introduce
739  // a problem at top and bottom boundaries
740  if (l_reflux && nrk == 2) {
741  int strt_comp_reflux = (l_const_rho) ? 1 : 0;
742  int num_comp_reflux = 1;
743  if (level < finest_level) {
744  fr_as_crse->CrseAdd(mfi,
745  {{AMREX_D_DECL(&(flux[0]), &(flux[1]), &(flux[2]))}},
746  dx, dt, strt_comp_reflux, strt_comp_reflux, num_comp_reflux, RunOn::Device);
747  }
748  if (level > 0) {
749  fr_as_fine->FineAdd(mfi,
750  {{AMREX_D_DECL(&(flux[0]), &(flux[1]), &(flux[2]))}},
751  dx, dt, strt_comp_reflux, strt_comp_reflux, num_comp_reflux, RunOn::Device);
752  }
753  } // two-way coupling
754  } // end profile
755  } // mfi
756  } // OMP
