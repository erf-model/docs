88 inline PlaneAverage::PlaneAverage(
89  const amrex::MultiFab* field_in,
90  amrex::Geometry geom_in,
91  int axis_in,
92  bool compute_deriv)
93  : m_field(field_in)
94  , m_geom(geom_in)
95  , m_axis(axis_in)
96  , m_comp_deriv(compute_deriv)
97 {
98  AMREX_ALWAYS_ASSERT(m_axis >= 0 && m_axis < AMREX_SPACEDIM);
99 
100  m_xlo = m_geom.ProbLo(m_axis);
101  m_dx = m_geom.CellSize(m_axis);
102  m_ncomp = m_field->nComp();
103 
104  amrex::Box domain = m_geom.Domain();
105  amrex::IntVect dom_lo(domain.loVect());
106  amrex::IntVect dom_hi(domain.hiVect());
107 
108  m_ncell_line = dom_hi[m_axis] - dom_lo[m_axis] + 1;
109 
110  m_ncell_plane = 1;
111  for (int i = 0; i < AMREX_SPACEDIM; ++i) {
112  if (i != m_axis) m_ncell_plane *= (dom_hi[i] - dom_lo[i] + 1);
113  }
114 
115  m_line_average.resize(static_cast<size_t>(m_ncell_line) * m_ncomp, 0.0);
116  if (m_comp_deriv)
117  m_line_deriv.resize(static_cast<size_t>(m_ncell_line) * m_ncomp, 0.0);
118  m_line_xcentroid.resize(m_ncell_line);
119 
120  for (int i = 0; i < m_ncell_line; ++i) {
121  m_line_xcentroid[i] = m_xlo + (i + 0.5) * m_dx;
122  }

125 inline amrex::Real
126 PlaneAverage::line_average_interpolated(amrex::Real x, int comp) const
127 {
128  AMREX_ALWAYS_ASSERT(comp >= 0 && comp < m_ncomp);
129 
130  amrex::Real c = 0.0;
131  int ind = 0;
132 
133  if (x > m_xlo + 0.5 * m_dx) {
134  ind = static_cast<int>(floor((x - m_xlo) / m_dx - 0.5));
135  const amrex::Real x1 = m_xlo + (ind + 0.5) * m_dx;
136  c = (x - x1) / m_dx;
137  }
138 
139  if (ind + 1 >= m_ncell_line) {
140  ind = m_ncell_line - 2;
141  c = 1.0;
142  }
143 
144  AMREX_ALWAYS_ASSERT(ind >= 0 && ind + 1 < m_ncell_line);
145 
146  return m_line_average[m_ncomp * ind + comp] * (1.0 - c) +
147  m_line_average[m_ncomp * (ind + 1) + comp] * c;
148 }

220 template <typename IndexSelector>
221 inline void PlaneAverage::compute_averages(
222  const IndexSelector& idxOp, const amrex::MultiFab& mfab)
223 {
224  const amrex::Real denom = 1.0 / (amrex::Real)m_ncell_plane;
225  amrex::AsyncArray<amrex::Real> lavg(m_line_average.data(), m_line_average.size());
226  amrex::Real* line_avg = lavg.data();
227  const int ncomp = m_ncomp;
228 
229 #ifdef _OPENMP
230 #pragma omp parallel if (Gpu::notInLaunchRegion())
231 #endif
232  for (amrex::MFIter mfi(mfab, amrex::TilingIfNotGPU()); mfi.isValid(); ++mfi) {
233  amrex::Box bx = mfi.tilebox();
234  auto fab_arr = mfab.const_array(mfi);
235  amrex::Box pbx = PerpendicularBox<IndexSelector>(bx, amrex::IntVect{0, 0, 0});
236 
237  ParallelFor(amrex::Gpu::KernelInfo().setReduction(true), pbx, [=]
238  AMREX_GPU_DEVICE( int p_i, int p_j, int p_k,
239  amrex::Gpu::Handler const& handler) noexcept {
240  // Loop over the direction perpendicular to the plane.
241  // This reduces the atomic pressure on the destination arrays.
242 
243  amrex::Box lbx = ParallelBox<IndexSelector>(bx, amrex::IntVect{p_i, p_j, p_k});
244 
245  for (int k = lbx.smallEnd(2); k <= lbx.bigEnd(2); ++k) {
246  for (int j = lbx.smallEnd(1); j <= lbx.bigEnd(1); ++j) {
247  for (int i = lbx.smallEnd(0); i <= lbx.bigEnd(0); ++i) {
248  int ind = idxOp.getIndx(i, j, k);
249  for (int n = 0; n < ncomp; ++n) {
250  amrex::Gpu::deviceReduceSum(&line_avg[ncomp * ind + n],
251  fab_arr(i, j, k, n) * denom, handler);
252  }
253  }
254  }
255  }
256  });
257  }
258 
259  lavg.copyToHost(m_line_average.data(), m_line_average.size());
260  amrex::ParallelDescriptor::ReduceRealSum(m_line_average.data(), m_line_average.size());
261 }
