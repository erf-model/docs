126 inline Real
127 PlaneAverage::line_average_interpolated(Real x, int comp) const
128 {
129  AMREX_ALWAYS_ASSERT(comp >= 0 && comp < m_ncomp);
130 
131  Real c = 0.0;
132  int ind = 0;
133 
134  if (x > m_xlo + 0.5 * m_dx) {
135  ind = static_cast<int>(floor((x - m_xlo) / m_dx - 0.5));
136  const Real x1 = m_xlo + (ind + 0.5) * m_dx;
137  c = (x - x1) / m_dx;
138  }
139 
140  if (ind + 1 >= m_ncell_line) {
141  ind = m_ncell_line - 2;
142  c = 1.0;
143  }
144 
145  AMREX_ALWAYS_ASSERT(ind >= 0 && ind + 1 < m_ncell_line);
146 
147  return m_line_average[m_ncomp * ind + comp] * (1.0 - c) +
148  m_line_average[m_ncomp * (ind + 1) + comp] * c;
149 }

221 template <typename IndexSelector>
222 inline void PlaneAverage::compute_averages(
223  const IndexSelector& idxOp, const MultiFab& mfab)
224 {
225  const Real denom = 1.0 / (Real)m_ncell_plane;
226  AsyncArray<Real> lavg(m_line_average.data(), m_line_average.size());
227  Real* line_avg = lavg.data();
228  const int ncomp = m_ncomp;
229 
230 #ifdef _OPENMP
231 #pragma omp parallel if (Gpu::notInLaunchRegion())
232 #endif
233  for (MFIter mfi(mfab, TilingIfNotGPU()); mfi.isValid(); ++mfi) {
234  Box bx = mfi.tilebox();
235  auto fab_arr = mfab.const_array(mfi);
236  Box pbx = PerpendicularBox<IndexSelector>(bx, IntVect{0, 0, 0});
237 
238  ParallelFor(Gpu::KernelInfo().setReduction(true), pbx, [=] AMREX_GPU_DEVICE( int p_i, int p_j, int p_k,
239  Gpu::Handler const& handler) noexcept {
240  // Loop over the direction perpendicular to the plane.
241  // This reduces the atomic pressure on the destination arrays.
242 
243  Box lbx = ParallelBox<IndexSelector>(bx, IntVect{p_i, p_j, p_k});
244 
245  for (int k = lbx.smallEnd(2); k <= lbx.bigEnd(2); ++k) {
246  for (int j = lbx.smallEnd(1); j <= lbx.bigEnd(1); ++j) {
247  for (int i = lbx.smallEnd(0); i <= lbx.bigEnd(0); ++i) {
248  int ind = idxOp.getIndx(i, j, k);
249  for (int n = 0; n < ncomp; ++n) {
250  Gpu::deviceReduceSum(&line_avg[ncomp * ind + n], fab_arr(i, j, k, n) * denom, handler);
251  }
252  }
253  }
254  }
255  });
256  }
257 
258  lavg.copyToHost(m_line_average.data(), m_line_average.size());
259  ParallelDescriptor::ReduceRealSum(m_line_average.data(), m_line_average.size());
260 }
