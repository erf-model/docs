104 inline amrex::Real
105 PlaneAverage::line_average_interpolated(amrex::Real x, int comp) const
106 {
107  AMREX_ALWAYS_ASSERT(comp >= 0 && comp < m_ncomp);
108 
109  amrex::Real c = 0.0;
110  int ind = 0;
111 
112  if (x > m_xlo + 0.5 * m_dx) {
113  ind = static_cast<int>(floor((x - m_xlo) / m_dx - 0.5));
114  const amrex::Real x1 = m_xlo + (ind + 0.5) * m_dx;
115  c = (x - x1) / m_dx;
116  }
117 
118  if (ind + 1 >= m_ncell_line) {
119  ind = m_ncell_line - 2;
120  c = 1.0;
121  }
122 
123  AMREX_ALWAYS_ASSERT(ind >= 0 && ind + 1 < m_ncell_line);
124 
125  return m_line_average[m_ncomp * ind + comp] * (1.0 - c) +
126  m_line_average[m_ncomp * (ind + 1) + comp] * c;
127 }
