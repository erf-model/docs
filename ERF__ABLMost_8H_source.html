470  int
471  lmask_min_reduce (amrex::iMultiFab& lmask, const int& nghost)
472  {
473  int lmask_min = amrex::ReduceMin(lmask, nghost,
474  [=] AMREX_GPU_HOST_DEVICE (amrex::Box const& bx, amrex::Array4<int const> const& lm_arr) -> int
475  {
476  int locmin = std::numeric_limits<int>::max();
477  const auto lo = lbound(bx);
478  const auto hi = ubound(bx);
479  for (int j = lo.y; j <= hi.y; ++j) {
480  for (int i = lo.x; i <= hi.x; ++i) {
481  locmin = std::min(locmin, lm_arr(i,j,0));
482  }
483  }
484  return locmin;
485  });
486 
487  return lmask_min;
488  }
