96 {
97  if ( k==izmax && c_ext_dir_on_zhi ) {
98  dthetadz = (1.0/3.0)*( -Thetav(i,j,k-1,cell_data,moist_flag)
99  - 3.0 * Thetav(i,j,k ,cell_data,moist_flag)
100  + 4.0 * Thetav(i,j,k+1,cell_data,moist_flag) )*dz_inv;
101  } else if ( k==izmin && c_ext_dir_on_zlo ) {
102  dthetadz = (1.0/3.0)*( Thetav(i,j,k+1,cell_data,moist_flag)
103  + 3.0 * Thetav(i,j,k ,cell_data,moist_flag)
104  - 4.0 * Thetav(i,j,k-1,cell_data,moist_flag) )*dz_inv;
105  } else {
106  dthetadz = 0.5*( Thetav(i,j,k+1,cell_data,moist_flag)
107  - Thetav(i,j,k-1,cell_data,moist_flag) )*dz_inv;
108  }
109 
110  if ( k==izmax && u_ext_dir_on_zhi ) {
111  dudz = (1.0/6.0)*( (-uvel(i ,j,k-1) - 3.0 * uvel(i ,j,k ) + 4.0 * uvel(i ,j,k+1))
112  + (-uvel(i+1,j,k-1) - 3.0 * uvel(i+1,j,k ) + 4.0 * uvel(i+1,j,k+1)) )*dz_inv;
113  } else if ( k==izmin && u_ext_dir_on_zlo ) {
114  dudz = (1.0/6.0)*( (uvel(i ,j,k+1) + 3.0 * uvel(i ,j,k ) - 4.0 * uvel(i ,j,k-1))
115  + (uvel(i+1,j,k+1) + 3.0 * uvel(i+1,j,k ) - 4.0 * uvel(i+1,j,k-1)) )*dz_inv;
116  } else {
117  dudz = 0.25*( uvel(i,j,k+1) - uvel(i,j,k-1) + uvel(i+1,j,k+1) - uvel(i+1,j,k-1) )*dz_inv;
118  }
119 
120  if ( k==izmax && v_ext_dir_on_zhi ) {
121  dvdz = (1.0/6.0)*( (-vvel(i,j ,k-1) - 3.0 * vvel(i,j ,k ) + 4.0 * vvel(i,j ,k+1))
122  + (-vvel(i,j+1,k-1) - 3.0 * vvel(i,j+1,k ) + 4.0 * vvel(i,j+1,k+1)) )*dz_inv;
123  } else if ( k==izmin && v_ext_dir_on_zlo ) {
124  dvdz = (1.0/6.0)*( (vvel(i,j ,k+1) + 3.0 * vvel(i,j ,k ) - 4.0 * vvel(i,j ,k-1))
125  + (vvel(i,j+1,k+1) + 3.0 * vvel(i,j+1,k ) - 4.0 * vvel(i,j+1,k-1)) )*dz_inv;
126  } else {
127  dvdz = 0.25*( vvel(i,j,k+1) - vvel(i,j,k-1) + vvel(i,j+1,k+1) - vvel(i,j+1,k-1) )*dz_inv;
128  }

182  // Note: Transport terms due to turbulence and pressure are included when
183  // DiffusionSrcForState_* is called from ERF_slow_rhs_post.
184 
185  // Shear Production (Note: We store mu_turb, which is 0.5*K_turb)
186  source_term += 4.0*K_turb(i,j,k,EddyDiff::Mom_v) * (dudz*dudz + dvdz*dvdz);

190

193  if (std::abs(qke) > 0.0) {
194  source_term -= 2.0 * cell_data(i,j,k,Rho_comp) * std::pow(qke,1.5) /
195  (pbl_mynn_B1_l * K_turb(i,j,k,EddyDiff::PBL_lengthscale));
196  }
