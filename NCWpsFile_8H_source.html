29 //
30 // NDArray is the datatype designed to hold any data, including scalars, multidimensional
31 // arrays, that read from the NetCDF file.
32 //
33 // The data read from NetCDF file are stored in a continuous memory, and the data layout is descriped
34 // by using a vector (shape). AMRex Box can be constructed using the data shape information, and MultiFab
35 // data array can be setup using the data that stored in the NDArray.
36 //
37 template <typename DataType>
38 struct NDArray
39 {
40  using DType = typename std::remove_const<DataType>::type;
41 
42  // constructor
43  explicit NDArray (const std::string vname, const std::vector<size_t>& vshape)
44  : name{vname}, shape{vshape}, ref_counted{1}, owned{true} {
45  data = new DType [this->ndim()];
46  }
47 
48  // default constructor
49  NDArray() : name{"null"}, data{nullptr}, ref_counted{1}, owned{false} {}
50 
51  // copy constructor
52  NDArray(const NDArray& array) {
53  name = array.name;
54  shape = array.shape;
55  data = array.data;
56  owned = false;
57  ref_counted.fetch_add(1, std::memory_order_relaxed);
58  }
59 
60  // copy assignment
61  NDArray& operator=(const NDArray& array) {
62  name = array.name;
63  shape = array.shape;
64  data = array.data;
65  owned = false;
66  ref_counted.fetch_add(1, std::memory_order_relaxed);
67  return *this;
68  }
69 
70  // destructor
71  ~NDArray() {
72  ref_counted.fetch_sub(1, std::memory_order_acq_rel);
73  if (ref_counted == 1 && owned) delete[] data;
74  }
75 
76  // get the data pointer
77  decltype(auto) get_data() {
78  ref_counted.fetch_add(1, std::memory_order_relaxed);
79  return data;
80  }
81 
82  // get the variable name
83  std::string get_vname() {
84  return name;
85  }
86 
87  // get the variable data shape
88  std::vector<size_t> get_vshape() {
89  return shape;
90  }
91 
92  // return the total number of data
93  size_t ndim() {
94  size_t num = 1;
95  for (auto i=0; i<shape.size(); ++i) num *= shape[i];
96  return num;
97  }
98

101  shape = vshape;
102  }
