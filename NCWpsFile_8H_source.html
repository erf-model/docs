47 //
48 // NDArray is the datatype designed to hold any data, including scalars, multidimensional
49 // arrays, that read from the NetCDF file.
50 //
51 // The data read from NetCDF file are stored in a continuous memory, and the data layout is descriped
52 // by using a vector (shape). AMRex Box can be constructed using the data shape information, and MultiFab
53 // data array can be setup using the data that stored in the NDArray.
54 //
55 template <typename DataType>
56 struct NDArray
57 {
58  using DType = typename std::remove_const<DataType>::type;
59 
60  // constructor
61  explicit NDArray (const std::string vname, const std::vector<size_t>& vshape)
62  : name{vname}, shape{vshape}, ref_counted{1}, owned{true} {
63  data = new DType [this->ndim()];
64  }
65 
66  // default constructor
67  NDArray() : name{"null"}, ref_counted{1}, owned{false}, data{nullptr} {}
68 
69  // copy constructor
70  NDArray(const NDArray& array) {
71  name = array.name;
72  shape = array.shape;
73  data = array.data;
74  owned = false;
75  ref_counted.fetch_add(1, std::memory_order_relaxed);
76  }
77 
78  // copy assignment

80  name = array.name;
81  shape = array.shape;

83  owned = false;
84  ref_counted.fetch_add(1, std::memory_order_relaxed);
85  return *this;
86  }
87 
88  // destructor
89  ~NDArray() {
90  ref_counted.fetch_sub(1, std::memory_order_acq_rel);
91  if (ref_counted == 1 && owned) delete[] data;
92  }
93
