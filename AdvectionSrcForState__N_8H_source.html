152  // NOTE: we don't need to weight avg_xmom, avg_ymom, avg_zmom with terrain metrics
153  // because that was done when they were constructed in AdvectionSrcForRhoAndTheta

161

163  interp_prim_h.InterpolateInY(i,j,k,prim_index,interpy_hi,interpy_lo,avg_ymom(i ,j+1,k ),avg_ymom(i ,j ,k ));
