1 /*-------------------------------------------------------------------------------------------
2  This subroutine is used to calculate droplet and ice crystal effective radius, which will be
3  used in the CAM radiation code. The method to calculate effective radius is taken out of the
4  Morrision two moment scheme from M2005MICRO_GRAUPEL. It is also very similar to the subroutine
5  effradius in the module of cldwat2m in the CAM source codes.
6  Adopted by Minghuai Wang (Minghuai.Wang@pnl.gov).
7  Calculate effective radius for radiation code
8  If no cloud water, default value is:
9  10 micron for droplets,
10  25 micron for cloud ice.
11  Be careful of the unit of effective radius : [micro meter]
12 
13  NOTE: this code is modified from E3SM
14 -------------------------------------------------------------------------------------------*/

53  // Main computation
54  const real pi = 3.1415926535897932384626434;
55  const real qsmall = 1.0e-14; // in the SAM source code (module_mp_graupel)
56  const real rhow = 997.; // in module_mp_graupel, SAM
57  const real rhoi = 500.; // in both CAM and SAM
58  const real dcs = 125.e-6; // in module_mp_graupel, SAM
59  const real ci = rhoi*pi/6.;
60  const real di = 3.;
61 
62  // for snow water
63  const real rhos = 100.;
64  const real cs = rhos*pi/6.;
65  const real ds = 3.;
66  const real mincld = 0.0001;
67 
68  int ncol = pres.extent(0);
69  int nlev = pres.extent(1);
70  // Effective diameters of snow crystals
71  parallel_for (SimpleBounds<2> (ncol, nlev), YAKL_LAMBDA (int i, int j) {
72  auto rho = pres(i,j)/(287.15*tk(i,j)); // air density [kg/m3]
73  auto cldm = std::max(cld(i,j), mincld);
74  auto qlic = std::min(5.e-3, std::max(0., ql(i,j)/cldm));
75  auto qiic = std::min(5.e-3, std::max(0., qi(i,j)/cldm));
76  auto nlic = std::max(nl(i,j), 0.)/cldm;
77  auto niic = std::max(ni(i,j), 0.)/cldm;
78 
79  real res;
80  if(qs(i,j) > 1.0e-7) {
81  auto lammaxs=1./10.e-6;
82  auto lammins=1./2000.e-6;
83  auto lams = std::pow((std::tgamma(1.+ds)*cs*ns(i,j)/qs(i,j)), (1./ds));
84  lams = std::min(lammaxs, std::max(lams,lammins));
85  res = 1.5/lams*1.0e6;
86  }
87  else {
88  res = 500.;
89  }
90 
91  // from Hugh Morrision: rhos/917 accouts for assumptions about
92  // ice density in the Mitchell optics.
93  des(i,j) = res*rhos/917.*2.;
94 
95  // Effective radius of cloud ice droplet
96  if( qiic >= qsmall ) {
97  niic = std::min(niic, qiic*1.e20);
98  auto lammax = 1./1.e-6; // in module_mp_graupel, SAM
99  auto lammin = 1./(2.*dcs+100.e-6); // in module_mp_graupel, SAM
100  auto lami = std::pow((std::tgamma(1.+di)*ci*niic/qiic), (1./di));
101  lami = std::min(lammax, std::max(lami, lammin));
102  effi(i,j) = 1.5/lami*1.e6;
103  }
104  else {
105  effi(i,j) = 25.;
106  }
107 
108  // hm ice effective radius for david mitchell's optics
109  // ac morrison indicates that this is effective diameter
110  // ac morrison indicates 917 (for the density of pure ice..)
111  deffi(i,j) = effi(i,j)*rhoi/917.*2.;
112 
113  // Effective radius of cloud liquid droplet
114  if( qlic >= qsmall ) {
115  // Matin et al., 1994 (JAS) formula for pgam (the same is used in both CAM and SAM).
116  // See also Morrison and Grabowski (2007, JAS, Eq. (2))
117  nlic = std::min(nlic, qlic*1.e20);
118 
119  // set the minimum droplet number as 20/cm3.
120  // nlic = max(nlic,20.e6_r8/rho) ! sghan minimum in #/cm3
121  auto tempnc = nlic/rho/1.0e6; // #/kg --> #/cm3
122 
123  // Should be the in-cloud dropelt number calculated as nlic*rho/1.0e6_r8 ????!!!! +++mhwang
124  auto pgam = 0.0005714*(nlic*rho/1.e6) + 0.2714;
125  pgam = 1./std::pow(pgam, 2.0)-1.;
126  pgam = std::min(10., std::max(pgam,2.));
127  auto laml = std::pow((pi/6.*rhow*nlic*std::tgamma(pgam+4.)/(qlic*std::tgamma(pgam+1.))), (1./3.));
128  auto lammin = (pgam+1.)/50.e-6; // in cldwat2m, CAM
129  auto lammax = (pgam+1.)/2.e-6; // if lammax is too large, this will lead to crash in
130  // src/physics/rrtmg/cloud_rad_props.F90 because
131  // klambda-1 can be zero in gam_liquid_lw and gam_liquid_sw
132  // and g_lambda(kmu,klambda-1) will not be defined.
133  laml = std::min(std::max(laml, lammin),lammax);
134  effl(i,j) = std::tgamma(pgam+4.)/std::tgamma(pgam+3.)/laml/2.*1.e6;
135  lamcrad(i,j) = laml;
136  pgamrad(i,j) = pgam;
137  }
138  else {
139  // we chose 10. over 25, since 10 is a more reasonable value for liquid droplet. +++mhwang
140  effl(i,j) = 10.; // in cldwat2m, CAM
141  lamcrad(i,j) = 0.0;
142  pgamrad(i,j) = 0.0;
143  }
144  });
