157  amrex::Real advance(T& S_old, T& S_new, amrex::Real time, const amrex::Real time_step)
158  {
159  timestep = time_step;
160  // Assume before advance() that S_old is valid data at the current time ("time" argument)
161  // And that if data is a MultiFab, both S_old and S_new contain ghost cells for evaluating a stencil based RHS
162  // We need this from S_old. This is convenient for S_new to have so we can use it
163  // as scratch space for stage values without creating a new scratch MultiFab with ghost cells.
164 
165  // Fill the RHS F_nodes at each stage
166  for (int i = 0; i < number_nodes; ++i)
167  {
168  // Get current stage time, t = t_old + h * Ci
169  amrex::Real stage_time = time + timestep * nodes[i];
170 
171  // Fill S_new with the solution value for evaluating F at the current stage
172  // Copy S_new = S_old
173  IntegratorOps<T>::Copy(S_new, S_old);
174  if (i > 0) {
175  // Saxpy across the tableau row:
176  // S_new += h * Aij * Fj
177  // We should fuse these kernels ...
178  for (int j = 0; j < i; ++j)
179  {
180  IntegratorOps<T>::Saxpy(S_new, timestep * tableau[i][j], *F_nodes[j]);
181  }
182 
183  // Call the post-update hook for the stage state value
184  BaseT::post_update(S_new, stage_time);
185  }
186 
187  // Fill F[i], the RHS at the current stage
188  // F[i] = RHS(y, t) at y = stage_value, t = stage_time
189  BaseT::rhs(*F_nodes[i], S_new, stage_time);
190  }
191 
192  // Fill new State, starting with S_new = S_old.
193  // Then Saxpy S_new += h * Wi * Fi for integration weights Wi
194  // We should fuse these kernels ...
195  IntegratorOps<T>::Copy(S_new, S_old);
196  for (int i = 0; i < number_nodes; ++i)
197  {
198  IntegratorOps<T>::Saxpy(S_new, timestep * weights[i], *F_nodes[i]);
199  }
200 
201  // Call the post-update hook for S_new
202  BaseT::post_update(S_new, time + timestep);
203 
204  // If we are working with an extended Butcher tableau, we can estimate the error here,
205  // and then calculate an adaptive timestep.
206 
207  // Return timestep
208  return timestep;
209  }
