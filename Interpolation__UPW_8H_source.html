594  AMREX_FORCE_INLINE
595  void
596  InterpolateInZ_hi(const int& i,
597  const int& j,
598  const int& k,
599  const int& qty_index,
600  amrex::Real& val_hi,
601  amrex::Real upw_hi,
602  const int& spatial_order) const
603  {
604  if (spatial_order == 2) {
605  val_hi = 0.5 * ( m_phi(i,j,k,qty_index) + m_phi(i,j,k+1,qty_index) );
606  return;
607  } else {
608  // Data to interpolate on
609  amrex::Real sp3 = (spatial_order > 4) ? m_phi(i , j , k+3, qty_index): 0.;
610  amrex::Real sp2 = m_phi(i , j , k+2, qty_index);
611  amrex::Real sp1 = m_phi(i , j , k+1, qty_index);
612  amrex::Real s = m_phi(i , j , k , qty_index);
613  amrex::Real sm1 = m_phi(i , j , k-1, qty_index);
614  amrex::Real sm2 = (spatial_order > 4) ? m_phi(i , j , k-2, qty_index) : 0.;
615 
616  // Upwinding flags
617  if (upw_hi != 0.) upw_hi = (upw_hi > 0) ? 1. : -1.;
618 
619  // Interpolate hi
620  val_hi = Evaluate(sp3,sp2,sp1,s,sm1,sm2,upw_hi,spatial_order);
621  }
622  }

624  AMREX_GPU_DEVICE
625  AMREX_FORCE_INLINE
626  amrex::Real
627  Evaluate(const amrex::Real& sp2,
628  const amrex::Real& sp1,
629  const amrex::Real& s,
630  const amrex::Real& sm1,
631  const amrex::Real& sm2,
632  const amrex::Real& sm3,
633  const amrex::Real& upw,
634  const int& spatial_order) const
635  {
636  // Averages and diffs
637  amrex::Real a1 = (s + sm1);
638  amrex::Real a2 = (sp1 + sm2);
639  amrex::Real d1 = (s - sm1);
640  amrex::Real d2 = (sp1 - sm2);
641  amrex::Real a3 = (sp2 + sm3);
642  amrex::Real d3 = (sp2 - sm3);
643 
644  // Interpolated value
645  amrex::Real iv;
646  switch (spatial_order) {
647  case 2:
648  iv = 0.5 * a1;
649  break;
650  case 3:
651  iv = g1_3_4*a1 - g2_3_4*a2 + upw * g2_3_4 * (d2 - 3.0*d1);
652  break;
653  case 4:
654  iv = g1_3_4*a1 - g2_3_4*a2;
655  break;
656  case 5:
657  iv = g1_5_6*a1 - g2_5_6*a2 + g3_5_6*a3 - upw * g3_5_6 * (d3 - 5.0*d2 + 10.0*d1);
658  break;
659  case 6:
660  iv = g1_5_6*a1 - g2_5_6*a2 + g3_5_6*a3;
661  break;
662  }
663  return ( iv );
664  }
