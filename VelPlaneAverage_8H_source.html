77  AMREX_ALWAYS_ASSERT(m_axis >= 0 && m_axis < AMREX_SPACEDIM);

79  m_dx = m_geom.CellSize(m_axis);

120 template <typename IndexSelector>
121 void VelPlaneAverage::compute_hvelmag_averages(
122  const IndexSelector& idx_op,
123  const int h1_idx,
124  const int h2_idx,
125  const Vector<MultiFab*> mfab)
126 {
127  const Real denom = 1.0 / (Real)m_ncell_plane;
128  AsyncArray<Real> lavg(m_line_hvelmag_average.data(), m_line_hvelmag_average.size());
129  Real* line_avg = lavg.data();
130 
131 #ifdef _OPENMP
132 #pragma omp parallel if (Gpu::notInLaunchRegion())
133 #endif
134  for (MFIter mfi(*(mfab[h1_idx]), TilingIfNotGPU()); mfi.isValid(); ++mfi) {
135  Box bx = mfi.tilebox();
136 
137  auto xfab_arr = mfab[h1_idx]->const_array(mfi);
138  auto yfab_arr = mfab[h2_idx]->const_array(mfi);
139 
140  Box pbx = PerpendicularBox<IndexSelector>(bx, IntVect{0, 0, 0});
141  ParallelFor(Gpu::KernelInfo().setReduction(true), pbx, [=] AMREX_GPU_DEVICE(
142  int p_i, int p_j, int p_k, Gpu::Handler const& handler) noexcept {
143  // Loop over the direction perpendicular to the plane.
144  // This reduces the atomic pressure on the destination arrays.
145  Box lbx = ParallelBox<IndexSelector>(bx, IntVect{p_i, p_j, p_k});
146 
147  for (int k = lbx.smallEnd(2); k <= lbx.bigEnd(2); ++k) {
148  for (int j = lbx.smallEnd(1); j <= lbx.bigEnd(1); ++j) {
149  for (int i = lbx.smallEnd(0); i <= lbx.bigEnd(0); ++i) {
150  const int ind = idx_op.getIndx(i,j,k);
151  const Real hvelmag = std::sqrt(xfab_arr(i,j,k)*xfab_arr(i,j,k)+yfab_arr(i,j,k)*yfab_arr(i,j,k));
152  Gpu::deviceReduceSum(&line_avg[ind],hvelmag*denom,handler);
153  }
154  }
155  }
156  });
157  }
158  lavg.copyToHost(m_line_hvelmag_average.data(), m_line_hvelmag_average.size());
159  ParallelDescriptor::ReduceRealSum(m_line_hvelmag_average.data(), m_line_hvelmag_average.size());
160 }

162 inline Real
163 VelPlaneAverage::line_hvelmag_average_interpolated(Real x) const
164 {
165  Real c = 0.0;
166  int ind = 0;
167  if (x > m_xlo + 0.5 * m_dx) {
168  ind = static_cast<int>(floor((x - m_xlo) / m_dx - 0.5));
169  const Real x1 = m_xlo + (ind + 0.5) * m_dx;
170  c = (x - x1) / m_dx;
171  }
172  if (ind + 1 >= m_ncell_line) {
173  ind = m_ncell_line - 2;
174  c = 1.0;
175  }
176  AMREX_ALWAYS_ASSERT(ind >= 0 && ind + 1 < m_ncell_line);
177  return m_line_hvelmag_average[ind] * (1.0 - c) +
178  m_line_hvelmag_average[ind + 1] * c;
179 }

189 VelPlaneAverage::line_hvelmag_derivative_of_average_cell(int ind) const
190 {
191  AMREX_ALWAYS_ASSERT(ind >= 0 && ind < m_ncell_line);
192  Real dudx;
193  if (ind == 0)
194  dudx =
195  (m_line_hvelmag_average[(ind + 1)] - m_line_hvelmag_average[ind]) /
196  m_dx;
197  else if (ind == m_ncell_line - 1)
198  dudx = (m_line_hvelmag_average[ind] - m_line_hvelmag_average[ind - 1]) /
199  m_dx;
200  else
201  dudx = 0.5 *
202  (m_line_hvelmag_average[ind + 1] -
203  m_line_hvelmag_average[ind - 1]) /
204  m_dx;
205 
206  return dudx;
