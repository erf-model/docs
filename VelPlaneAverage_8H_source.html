65 inline VelPlaneAverage::VelPlaneAverage(
66  const amrex::Vector<amrex::MultiFab*>& mfab_in,
67  amrex::Geometry geom_in,
68  int axis_in,
69  bool compute_deriv)
70  : m_mfab(mfab_in)
71  , m_geom(geom_in)
72  , m_axis(axis_in)
73  , m_comp_deriv(compute_deriv) {
74  AMREX_ALWAYS_ASSERT(m_axis >= 0 && m_axis < AMREX_SPACEDIM);
75  m_xlo = m_geom.ProbLo(m_axis);
76  m_dx = m_geom.CellSize(m_axis);
77 
78  amrex::Box domain = m_geom.Domain();
79  amrex::IntVect dom_lo(domain.loVect());
80  amrex::IntVect dom_hi(domain.hiVect());
81 
82  m_ncell_line = dom_hi[m_axis] - dom_lo[m_axis] + 1;
83  m_ncell_plane = 1;
84  for (int i = 0; i < AMREX_SPACEDIM; ++i) {
85  if (i != m_axis) m_ncell_plane *= (dom_hi[i] - dom_lo[i] + 1);
86  }
87 
88  m_line_xcentroid.resize(m_ncell_line);
89  for (int i = 0; i < m_ncell_line; ++i) {
90  m_line_xcentroid[i] = m_xlo + (i + 0.5) * m_dx;
91  }
92 
93  m_line_hvelmag_average.resize(m_ncell_line, 0.0);
94  if (m_comp_deriv) m_line_hvelmag_deriv.resize(m_ncell_line, 0.0);
95 }

117 template <typename IndexSelector>
118 void VelPlaneAverage::compute_hvelmag_averages(
119  const IndexSelector& idx_op,
120  const int h1_idx,
121  const int h2_idx,
122  const amrex::Vector<amrex::MultiFab*> mfab)
123 {
124  const amrex::Real denom = 1.0 / (amrex::Real)m_ncell_plane;
125  amrex::AsyncArray<amrex::Real> lavg(m_line_hvelmag_average.data(), m_line_hvelmag_average.size());
126  amrex::Real* line_avg = lavg.data();
127 
128 #ifdef _OPENMP
129 #pragma omp parallel if (Gpu::notInLaunchRegion())
130 #endif
131  for (amrex::MFIter mfi(*(mfab[h1_idx]), amrex::TilingIfNotGPU()); mfi.isValid(); ++mfi) {
132  amrex::Box bx = mfi.tilebox();
133 
134  auto xfab_arr = mfab[h1_idx]->const_array(mfi);
135  auto yfab_arr = mfab[h2_idx]->const_array(mfi);
136 
137  amrex::Box pbx = PerpendicularBox<IndexSelector>(bx, amrex::IntVect{0, 0, 0});
138  ParallelFor(amrex::Gpu::KernelInfo().setReduction(true), pbx, [=] AMREX_GPU_DEVICE(
139  int p_i, int p_j, int p_k, amrex::Gpu::Handler const& handler) noexcept {
140  // Loop over the direction perpendicular to the plane.
141  // This reduces the atomic pressure on the destination arrays.
142  amrex::Box lbx = ParallelBox<IndexSelector>(bx, amrex::IntVect{p_i, p_j, p_k});
143 
144  for (int k = lbx.smallEnd(2); k <= lbx.bigEnd(2); ++k) {
145  for (int j = lbx.smallEnd(1); j <= lbx.bigEnd(1); ++j) {
146  for (int i = lbx.smallEnd(0); i <= lbx.bigEnd(0); ++i) {
147  const int ind = idx_op.getIndx(i,j,k);
148  const amrex::Real hvelmag =
149  std::sqrt(xfab_arr(i,j,k)*xfab_arr(i,j,k)+yfab_arr(i,j,k)*yfab_arr(i,j,k));
150  amrex::Gpu::deviceReduceSum(&line_avg[ind],hvelmag*denom,handler);
151  }
152  }
153  }
154  });
155  }
156  lavg.copyToHost(m_line_hvelmag_average.data(), m_line_hvelmag_average.size());
157  amrex::ParallelDescriptor::ReduceRealSum(m_line_hvelmag_average.data(), m_line_hvelmag_average.size());
158 }

160 inline amrex::Real
161 VelPlaneAverage::line_hvelmag_average_interpolated(amrex::Real x) const
162 {
163  amrex::Real c = 0.0;
164  int ind = 0;
165  if (x > m_xlo + 0.5 * m_dx) {
166  ind = static_cast<int>(floor((x - m_xlo) / m_dx - 0.5));
167  const amrex::Real x1 = m_xlo + (ind + 0.5) * m_dx;
168  c = (x - x1) / m_dx;
169  }
170  if (ind + 1 >= m_ncell_line) {
171  ind = m_ncell_line - 2;
172  c = 1.0;
173  }
174  AMREX_ALWAYS_ASSERT(ind >= 0 && ind + 1 < m_ncell_line);
175  return m_line_hvelmag_average[ind] * (1.0 - c) +
176  m_line_hvelmag_average[ind + 1] * c;
177 }

186 inline amrex::Real
187 VelPlaneAverage::line_hvelmag_derivative_of_average_cell(int ind) const
188 {
189  AMREX_ALWAYS_ASSERT(ind >= 0 && ind < m_ncell_line);
190  amrex::Real dudx;
191  if (ind == 0)
192  dudx =
193  (m_line_hvelmag_average[(ind + 1)] - m_line_hvelmag_average[ind]) /
194  m_dx;
195  else if (ind == m_ncell_line - 1)
196  dudx = (m_line_hvelmag_average[ind] - m_line_hvelmag_average[ind - 1]) /
197  m_dx;
198  else
199  dudx = 0.5 *
200  (m_line_hvelmag_average[ind + 1] -
201  m_line_hvelmag_average[ind - 1]) /
202  m_dx;
203 
204  return dudx;
205 }
