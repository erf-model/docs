35  {
36  if (zeta > 0) {
37  return -beta_m * zeta;
38  } else {
39  amrex::Real x = std::sqrt(std::sqrt(1.0 - gamma_m * zeta));
40  return 2.0 * std::log(0.5 * (1.0 + x)) + log(0.5 * (1.0 + x * x)) -
41  2.0 * std::atan(x) + PIoTwo;
42  }
43  }

535  AMREX_FORCE_INLINE
536  void
537  iterate_flux (const int& i,
538  const int& j,
539  const int& k,
540  const int& max_iters,
541  const amrex::Array4<amrex::Real>& z0_arr,
542  const amrex::Array4<const amrex::Real>& umm_arr,
543  const amrex::Array4<const amrex::Real>& tm_arr,
544  const amrex::Array4<amrex::Real>& u_star_arr,
545  const amrex::Array4<amrex::Real>& t_star_arr,
546  const amrex::Array4<amrex::Real>& t_surf_arr,
547  const amrex::Array4<amrex::Real>& olen_arr) const
548  {
549  int iter = 0;
550  amrex::Real ustar = 0.0;
551  amrex::Real z0 = 0.0;
552  amrex::Real tflux = 0.0;
553  amrex::Real zeta = 0.0;
554  amrex::Real psi_m = 0.0;
555  amrex::Real psi_h = 0.0;
556  amrex::Real Olen = 0.0;
557  u_star_arr(i,j,k) = mdata.kappa * umm_arr(i,j,k) / std::log(mdata.zref / z0_arr(i,j,k));
558  do {
559  ustar = u_star_arr(i,j,k);
560  z0 = std::exp( (2.7*ustar - 1.8/mdata.Cnk_b) / (ustar + 0.17/mdata.Cnk_b) );
561  tflux = -(tm_arr(i,j,k) - t_surf_arr(i,j,k)) * ustar * mdata.kappa /
562  (std::log(mdata.zref / z0) - psi_h);
563  Olen = -ustar * ustar * ustar * tm_arr(i,j,k) /
564  (mdata.kappa * mdata.gravity * tflux);
565  zeta = mdata.zref / Olen;
566  psi_m = sfuns.calc_psi_m(zeta);
567  psi_h = sfuns.calc_psi_h(zeta);
568  u_star_arr(i,j,k) = mdata.kappa * umm_arr(i,j,k) / (std::log(mdata.zref / z0) - psi_m);
569  ++iter;
570  } while ((std::abs(u_star_arr(i,j,k) - ustar) > tol) && iter <= max_iters);
571 
572  t_star_arr(i,j,k) = mdata.kappa * (tm_arr(i,j,k) - t_surf_arr(i,j,k)) /
573  (std::log(mdata.zref / z0) - psi_h);
574  olen_arr(i,j,k) = Olen;
575  z0_arr(i,j,k) = z0;
576  }
