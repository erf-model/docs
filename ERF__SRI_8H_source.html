146  amrex::Real advance (T& S_old, T& S_new, amrex::Real time, const amrex::Real time_step)
147  {
148  // Assume before advance() that S_old is valid data at the current time ("time" argument)
149  // And that both S_old and S_new contain ghost cells for evaluating a stencil based RHS
150 
151  // **********************************************
152  // RK Integration
153  // **********************************************
154 
155  // We copy all variables when we call Copy in the following -- the velocities
156  // will be over-written in the rhs on all valid faces but we
157  // use this copy to fill in the ghost locations which will
158  // be needed for metric terms
159 
160  timestep = time_step;
161 
162  int nav = Cons::NumVars;
163  const amrex::Vector<int> scomp_all = {0,0,0,0,0,0,0};
164  const amrex::Vector<int> ncomp_all = {nav,1,1,1,nav,nav,nav};
165 
166  amrex::Vector<amrex::MultiFab> S_scratch;
167 
168  // Fill the RHS F_nodes at each stage
169  for (int i = 0; i < number_nodes; ++i)
170  {
171  // Get current stage time, t = t_old + h * Ci
172  amrex::Real stage_time = time + time_step * nodes[i];
173 
174  // Fill S_new with the solution value for evaluating F at the current stage
175  // Copy S_new = S_old
176  amrex::IntegratorOps<T>::Copy(S_new, S_old);
177  if (i > 0) {
178  // Saxpy across the tableau row:
179  // S_new += h * Aij * Fj
180  // We should fuse these kernels ...
181  for (int j = 0; j < i; ++j)
182  {
183  amrex::IntegratorOps<T>::Saxpy(S_new, time_step * tableau[i][j], *F_nodes[j], scomp_all, ncomp_all);
184  }
185 
186  // Call the post-update hook for the stage state value
187  post_update(S_new, stage_time, S_new[IntVar::cons].nGrow(), S_new[IntVar::xmom].nGrow());
188  }
189 
190  // Fill F[i], the RHS at the current stage
191  // F[i] = RHS(y, t) at y = stage_value, t = stage_time
192  rhs(*F_nodes[i], S_new, S_scratch, stage_time, RHSVar::all);
193  }
194 
195  // Fill new State, starting with S_new = S_old.
196  // Then Saxpy S_new += h * Wi * Fi for integration weights Wi
197  // We should fuse these kernels ...
198  amrex::IntegratorOps<T>::Copy(S_new, S_old);
199  for (int i = 0; i < number_nodes; ++i)
200  {
201  amrex::IntegratorOps<T>::Saxpy(S_new, time_step * weights[i], *F_nodes[i], scomp_all, ncomp_all);
202  }
203 
204  // Call the post-update hook for S_new
205  post_update(S_new, time + time_step, S_new[IntVar::cons].nGrow(), S_new[IntVar::xmom].nGrow());
206 
207  // Return timestep
208  return time_step;
209  }
