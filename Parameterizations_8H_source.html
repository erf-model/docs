13  real liquid_effective_radius(real temperature, real landfrac, real icefrac = 0., real snowh = 0.) {
14  const real rliqland = 8.0; // liquid drop size if over land
15  const real rliqocean = 14.0; // liquid drop size if over ocean
16  const real rliqice = 14.0; // liquid drop size if over sea ice
17 
18  real reffective = 0.;
19  // jrm Reworked effective radius algorithm
20  // Start with temperature-dependent value appropriate for continental air
21  reffective = rliqland+(rliqocean-rliqland)*std::min(1.0, std::max(0.0, (tmelt-temperature)*0.05));
22 
23  // Modify for snow depth over land
24  reffective = reffective + (rliqocean-reffective)*std::min(1.0, std::max(0.0, snowh*10.));
25 
26  // Ramp between polluted value over land to clean value over ocean.
27  reffective = reffective + (rliqocean-reffective) * std::min(1.0, std::max(0.0, 1.0 - landfrac));
28 
29  // Ramp between the resultant value and a sea ice value in the presence of ice.
30  reffective = reffective + (rliqice-reffective) * std::min(1.0, std::max(0.0, icefrac));
31  return reffective;
32  }

42  // Computes surface albedos over ocean and the surface
43  // Two spectral surface albedos for direct (dir) and diffuse (dif)
44  // incident radiation are calculated. The spectral intervals are:
45  // s (shortwave) = 0.2-0.7 micro-meters
46  // l (longwave) = 0.7-5.0 micro-meters
47  //
48  // Uses knowledge of surface type to specify albedo, as follows:
49  //
50  // Ocean Uses solar zenith angle to compute albedo for direct
51  // radiation; diffuse radiation values constant; albedo
52  // independent of spectral interval and other physical
53  // factors such as ocean surface wind speed.
54  //
55  // For more details , see Briegleb, Bruce P., 1992: Delta-Eddington
56  // Approximation for Solar Radiation in the NCAR Community Climate Model,
57  // Journal of Geophysical Research, Vol 97, D7, pp7603-7612).
58  AMREX_FORCE_INLINE
59  void albedo(bool ocean, real1d& coszrs, real1d& tsurface, real1d& asdir,
60  real1d& aldir, real1d& asdif, real1d& aldif)
61  {
62  using yakl::intrinsics::size;
63  const real adif = 0.06;
64  int ncol = size(tsurface, 1);
65  if (ocean) {
66  // Ice-free ocean albedos function of solar zenith angle only, and
67  // independent of spectral interval:
68  yakl::c::parallel_for(ncol, YAKL_LAMBDA (int icol) {
69  if (coszrs(icol) <= 0) {
70  aldir(icol) = 0;
71  asdir(icol) = 0;
72  aldif(icol) = 0;
73  asdif(icol) = 0;
74  } else if (tsurface(icol) > 271.) {
75  aldir(icol) = ( .026 / (std::pow(coszrs(icol), 1.7) + .065)) +
76  (.15*(coszrs(icol) - 0.10) * (coszrs(icol) - 0.50) * (coszrs(icol) - 1.00) );
77  asdir(icol) = aldir(icol);
78  aldif(icol) = adif;
79  asdif(icol) = adif;
80  } else {
81  // albedo of sea-ice/snow surface
82  aldir(icol) = 0.45;
83  asdir(icol) = 0.75;
84  aldif(icol) = 0.45;
85  asdif(icol) = 0.75;
86  }
87  });
88  } else {// land
89  yakl::c::parallel_for(ncol, YAKL_LAMBDA (int icol) {
90  if (coszrs(icol) <= 0.) {
91  aldir(icol) = 0.;
92  asdir(icol) = 0.;
93  aldif(icol) = 0.;
94  asdif(icol) = 0.;
95  } else {
96  // Albedos for land type I (Briegleb)
97  asdir(icol) = 1.4 * 0.06 / ( 1. + 0.8 * coszrs(icol));
98  asdif(icol) = 1.2 * 0.06;
99  aldir(icol) = 1.4 * 0.24 / ( 1. + 0.8 * coszrs(icol));
100  aldif(icol) = 1.2 * 0.24;
101  }
102  });
103  }
