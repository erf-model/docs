101 calc_rho_p (const int& kmax,
102  const int& flag_psfc,
103  const amrex::Real& psfc,
104  amrex::Real* Thetad_vec,
105  amrex::Real* Thetam_vec,
106  amrex::Real* Q_vec,
107  amrex::Real* z_vec,
108  amrex::Real* Rhod_vec,
109  amrex::Real* Rhom_vec,
110  amrex::Real* Pd_vec,
111  amrex::Real* Pm_vec)
112 {
113  const int maxiter = 10;
114 
115  // Calculate or use moist pressure at the surface.
116  if (flag_psfc == 1) {
117  Pm_vec[0] = psfc;
118  } else {
119  amrex::Real t_0 = 290.0; // WRF's model_config_rec%base_temp
120  amrex::Real a = 50.0; // WRF's model_config_rec%base_lapse
121  Pm_vec[0] = p_0*exp(-t_0/a+std::pow((std::pow(t_0/a, 2)-2.0*CONST_GRAV*z_vec[0]/(a*R_d)), 0.5));
122  }
123 
124  // calculate virtual potential temperature at the surface.
125  {
126  amrex::Real qvf = 1.0+(R_v/R_d+1.0)*Q_vec[0];
127  Thetam_vec[0] = Thetad_vec[0]*qvf;
128  }
129 
130  // calculate moist density at the surface.
131  Rhom_vec[0] = 1.0/(R_d/p_0*Thetam_vec[0]*std::pow(Pm_vec[0]/p_0, -iGamma));
132 
133  // integrate from the surface to the top boundary.
134  for (int k=1; k<=kmax; k++) {
135  amrex::Real dz = z_vec[k]-z_vec[k-1];
136  amrex::Real qvf = 1.0+(R_v/R_d+1.0)*Q_vec[k];
137  Thetam_vec[k] = Thetad_vec[k]*qvf;
138  Rhom_vec[k] = Rhom_vec[k-1]; // an initial guess.
139  for (int it=0; it<maxiter; it++) {
140  Pm_vec[k] = Pm_vec[k-1]-0.5*dz*(Rhom_vec[k]+Rhom_vec[k-1])*CONST_GRAV;
141  if (Pm_vec[k] < 0.0) Pm_vec[k] = 0.0;
142  Rhom_vec[k] = 1.0/(R_d/p_0*Thetam_vec[k]*std::pow(Pm_vec[k]/p_0, -iGamma));
143  } // it
144  } // k
145 
146  // integrate from the top back down to get dry pressure and density.
147  Pd_vec[kmax] = Pm_vec[kmax];
148  Rhod_vec[kmax] = 1.0/(R_d/p_0*Thetam_vec[kmax]*std::pow(Pd_vec[kmax]/p_0, -iGamma));
149  for (int k=kmax-1; k>=0; k--) {
150  amrex::Real dz = z_vec[k+1]-z_vec[k];
151  Rhod_vec[k] = Rhod_vec[k+1]; // an initial guess.
152  for (int it=0; it < maxiter; it++) {
153  Pd_vec[k] = Pd_vec[k+1]+0.5*dz*(Rhod_vec[k]+Rhod_vec[k+1])*CONST_GRAV;
154  if (Pd_vec[k] < 0.0) Pd_vec[k] = 0.0;
155  Rhod_vec[k] = 1.0/(R_d/p_0*Thetam_vec[k]*std::pow(Pd_vec[k]/p_0, -iGamma));
156  } // it
157  } // k
158 }

287 rh_to_mxrat (int i,
288  int j,
289  int k,
290  const amrex::Array4<amrex::Real const>& rhum,
291  const amrex::Array4<amrex::Real const>& temp,
292  const amrex::Array4<amrex::Real const>& pres,
293  const amrex::Array4<amrex::Real>& mxrat)
294 {
295  amrex::Real qv_max_p_safe = 10000.0; // WRF default value
296  amrex::Real qv_max_flag = 1.0e-5; // WRF default value
297  amrex::Real qv_max_value = 3.0e-6; // WRF default value
298  amrex::Real qv_min_p_safe = 110000.0; // WRF default value
299  amrex::Real qv_min_flag = 1.0e-6; // WRF default value
300  amrex::Real qv_min_value = 1.0e-6; // WRF default value
301  amrex::Real eps = 0.622;
302  amrex::Real svp1 = 0.6112;
303  amrex::Real svp2 = 17.67;
304  amrex::Real svp3 = 29.65;
305  amrex::Real svpt0 = 273.15;
306  // WRF's method when model_config_rec%rh2qv_wrt_liquid=.true. (default behavior)
307  if (temp(i,j,k) != 0.0) {
308  amrex::Real es=0.01*rhum(i,j,k)*svp1*10.0*exp(svp2*(temp(i,j,k)-svpt0)/(temp(i,j,k)-svp3));
309  if (es >= pres(i,j,k)/100.0) {
310  // vapor pressure exceeds total pressure
311  mxrat(i,j,k) = std::pow(10.0, -6);
312  }
313  else {
314  mxrat(i,j,k) = amrex::max(eps*es/(pres(i,j,k)/100.0-es), 1.0e-6);
315  }
316  }
317  else {
318  // I don't know why there's a fringe case handled in WRF where T is absolute zero...
319  // Let's just deal with it here in case we also end up needing it.
320  mxrat(i,j,k) = 1.0e-6;
321  }
322  // See the below comment from WRF dyn_em/module_initialize_real.F rh_to_mxrat1.
323  // For pressures above a defined level, reasonable Qv values should be
324  // a certain value or smaller. If they are larger than this, the input data
325  // probably had "missing" RH, and we filled in some values. This is an
326  // attempt to catch those. Also, set the minimum value for the entire
327  // domain that is above the selected pressure level.
328  if (pres(i,j,k) < qv_max_p_safe) {
329  if (mxrat(i,j,k) > qv_max_flag) {
330  mxrat(i,j,k) = qv_max_value;
331  }
332  }
333  if (pres(i,j,k) < qv_min_p_safe) {
334  if (mxrat(i,j,k) < qv_min_flag) {
335  mxrat(i,j,k) = qv_min_value;
336  }
337  }
338 }
